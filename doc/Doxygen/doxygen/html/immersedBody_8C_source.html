185 void immersedBody::constructRefineField
186 (
187  volScalarField& body,
188  volScalarField& refineF,
189  DynamicLabelList cellsToIterate,
190  DynamicLabelList startLevel
191 )
192 {
193  if(refineBuffers_ == 0)
194  return;
195 
196  DynamicLabelList cellsToIterateC;
197  DynamicLabelList cellsToIterateF;
198 
199  List<DynamicLabelList> cellsToSendToProcs;
200  cellsToSendToProcs.setSize(Pstream::nProcs());
201  List<DynamicLabelList> cellsToSendToProcsLevel;
202  cellsToSendToProcsLevel.setSize(Pstream::nProcs());
203 
204  for(label i = 0; i < refineBuffers_; i++)
205  {
206  forAll(cellsToIterate, cellI)
207  {
208  if(startLevel[cellI] == i)
209  cellsToIterateC.append(cellsToIterate[cellI]);
210  }
211 
212  forAll(cellsToIterateC, cellI)
213  {
214  labelList cellFaces(mesh_.cells()[cellsToIterateC[cellI]]);
215  forAll(cellFaces, faceI)
216  {
217  if (mesh_.isInternalFace(cellFaces[faceI]))
218  {
219  label nCell(mesh_.owner()[cellFaces[faceI]]);
220  if(nCell == cellsToIterateC[cellI])
221  {
222  nCell = mesh_.neighbour()[cellFaces[faceI]];
223  }
224 
225  if(refineF[nCell] == 0)
226  {
227  if(i > 0)
228  {
229  if(body[nCell] < SMALL)
230  {
231  refineF[nCell] = 1;
232  cellsToIterateF.append(nCell);
233  }
234  }
235  else
236  {
237  refineF[nCell] = 1;
238  cellsToIterateF.append(nCell);
239  }
240  }
241  }
242  else
243  {
244  label facePatchId(mesh_.boundaryMesh().whichPatch(
245  cellFaces[faceI]
246  ));
247 
248  const polyPatch& cPatch = mesh_.boundaryMesh()[facePatchId];
249  if (cPatch.type() == "processor")
250  {
251  const processorPolyPatch& procPatch =
252  refCast<const processorPolyPatch>(cPatch);
253  if (procPatch.myProcNo() == Pstream::myProcNo())
254  {
255  cellsToSendToProcs[procPatch.neighbProcNo()].append(
256  cPatch.whichFace(cellFaces[faceI])
257  );
258  cellsToSendToProcsLevel[procPatch.neighbProcNo()]
259  .append(i+1);
260  }
261  else
262  {
263  cellsToSendToProcs[procPatch.myProcNo()].append(
264  cPatch.whichFace(cellFaces[faceI])
265  );
266  cellsToSendToProcsLevel[procPatch.myProcNo()]
267  .append(i+1);
268  }
269  }
270  }
271  }
272  }
273  cellsToIterateC = cellsToIterateF;
274  cellsToIterateF.clear();
275  }
276 
277  List<DynamicLabelList> facesReceivedFromProcs;
278  List<DynamicLabelList> cellsReceivedFromProcsLevel;
279 
280  // send points that are not on this proc to other proc
281  PstreamBuffers pBufs(Pstream::commsTypes::nonBlocking);
282  for (label proci = 0; proci < Pstream::nProcs(); proci++)
283  {
284  if (proci != Pstream::myProcNo())
285  {
286  UOPstream send(proci, pBufs);
287  send << cellsToSendToProcs[proci];
288  }
289  }
290  pBufs.finishedSends();
291  // recieve points from other procs
292  for (label proci = 0; proci < Pstream::nProcs(); proci++)
293  {
294  if (proci != Pstream::myProcNo())
295  {
296  UIPstream recv(proci, pBufs);
297  DynamicLabelList recList (recv);
298  facesReceivedFromProcs.append(recList);
299  }
300  else
301  {
302  DynamicLabelList recList;
303  facesReceivedFromProcs.append(recList);
304  }
305  }
306  pBufs.clear();
307 
308  for (label proci = 0; proci < Pstream::nProcs(); proci++)
309  {
310  if (proci != Pstream::myProcNo())
311  {
312 
313  UOPstream send(proci, pBufs);
314  send << cellsToSendToProcsLevel[proci];
315  }
316  }
317  pBufs.finishedSends();
318 
319  // recieve points from other procs
320  for (label proci = 0; proci < Pstream::nProcs(); proci++)
321  {
322  if (proci != Pstream::myProcNo())
323  {
324  UIPstream recv(proci, pBufs);
325  DynamicLabelList recList (recv);
326  cellsReceivedFromProcsLevel.append(recList);
327  }
328  else
329  {
330  DynamicLabelList recList;
331  cellsReceivedFromProcsLevel.append(recList);
332  }
333  }
334  pBufs.clear();
335 
336 
337  DynamicLabelList newCellsToIterate;
338  DynamicLabelList newCellsToIterateStartLevel;
339 
340  // check if some point from other proc is on this processor
341  for (label otherProci = 0;
342  otherProci < facesReceivedFromProcs.size();
343  otherProci++
344  )
345  {
346  for (label faceI = 0;
347  faceI < facesReceivedFromProcs[otherProci].size();
348  faceI++
349  )
350  {
351  label cellProcI(0);
352  forAll (mesh_.boundaryMesh(), patchi)
353  {
354  const polyPatch& cPatch = mesh_.boundaryMesh()[patchi];
355  if (cPatch.type() == "processor")
356  {
357  const processorPolyPatch& procPatch =
358  refCast<const processorPolyPatch>(cPatch);
359 
360  if (procPatch.myProcNo() == Pstream::myProcNo()
361  && procPatch.neighbProcNo() == otherProci)
362  {
363  cellProcI = mesh_.faceOwner()[cPatch.start()
364  + facesReceivedFromProcs[otherProci][faceI]];
365  if(refineF[cellProcI] == 0)
366  {
367  newCellsToIterate.append(cellProcI);
368  newCellsToIterateStartLevel.append(
369  cellsReceivedFromProcsLevel[otherProci][faceI]
370  );
371  }
372  break;
373  }
374  else if (procPatch.myProcNo() == otherProci
375  && procPatch.neighbProcNo() == Pstream::myProcNo())
376  {
377  cellProcI = mesh_.faceOwner()[cPatch.start()
378  + facesReceivedFromProcs[otherProci][faceI]];
379  if(refineF[cellProcI] == 0)
380  {
381  newCellsToIterate.append(cellProcI);
382  newCellsToIterateStartLevel.append(
383  cellsReceivedFromProcsLevel[otherProci][faceI]
384  );
385  }
386  break;
387  }
388  }
389  }
390  }
391  }
392 
393  bool contBool(false);
394  if(newCellsToIterate.size() > 0)
395  {
396  contBool = true;
397  }
398 
399  reduce(contBool, orOp<bool>());
400 
401  if(contBool)
402  {
403  constructRefineField
404  (
405  body,
406  refineF,
407  newCellsToIterate,
408  newCellsToIterateStartLevel
409  );
410  }

427 void immersedBody::updateCoupling
428 (
429  volScalarField& body,
430  volVectorField& f
431 )
432 {
433  vector FV(vector::zero);
434  vector TA(vector::zero);
435 
436  List<DynamicLabelList> intLists;
437  List<DynamicLabelList> surfLists;
438  DynamicVectorList refCoMList;
439 
440  geomModel_->getReferencedLists(
441  intLists,
442  surfLists,
443  refCoMList
444  );
445 
446  // calcualate viscous force and torque
447 
448  forAll (intLists, i)
449  {
450  DynamicLabelList& intListI = intLists[i];
451  forAll (intListI, intCell)
452  {
453  label cellI = intListI[intCell];
454 
455  FV -= f[cellI]*mesh_.V()[cellI];
456  TA -= ((mesh_.C()[cellI] - refCoMList[i])^f[cellI])
457  *mesh_.V()[cellI];
458  }
459  }
460 
461  forAll (surfLists, i)
462  {
463  DynamicLabelList& surfListI = surfLists[i];
464  forAll (surfListI, surfCell)
465  {
466  label cellI = surfListI[surfCell];
467 
468  FV -= f[cellI]*mesh_.V()[cellI];
469  TA -= ((mesh_.C()[cellI] - refCoMList[i])^f[cellI])
470  *mesh_.V()[cellI];
471  }
472  }
473 
474  reduce(FV, sumOp<vector>());
475  reduce(TA, sumOp<vector>());
476  FV *= rhoF_.value();
477  TA *= rhoF_.value();
478 
479  FCoupling_ = forces(FV, TA);
480 }

500 void immersedBody::updateMovementComp
501 (
502  scalar deltaT,
503  vector Vel,
504  vector Axis,
505  scalar omega
506 )
507 {
508  auto updateTranslation = [&]()
509  {
510 
511  const uniformDimensionedVectorField& g =
512  mesh_.lookupObject<uniformDimensionedVectorField>("g");
513  vector FG(geomModel_->getM0()*(1.0-rhoF_.value()
514  /geomModel_->getRhoS().value())*g.value());
515 
516  vector F(FCoupling_.F);
517  F += FContact_.F;
518  F += FG;
519 
520  if(!case3D)
521  {
522  F[emptyDim] *= 0;
523  FG[emptyDim] *= 0;
524  }
525  if(geomModel_->getM0() > 0)
526  {
527  // compute current acceleration (assume constant over timeStep)
528 
529  InfoH << iB_Info <<"-- body "<< bodyId_ <<" ParticelMass : " << geomModel_->getM0() << endl;
530  InfoH << iB_Info <<"-- body "<< bodyId_ <<" Acting Force : " << F << endl;
531  a_ = F/(geomModel_->getM0());
532  // update body linear velocity
533  Vel_ = Vel + deltaT*a_;
534  InfoH << iB_Info <<"-- body "<< bodyId_ <<" accelaration : " << a_ << endl;
535  }
536  };
537 
538  auto updateRotation = [&]()
539  {
540  if(mag(geomModel_->getI()) > 0)
541  {
542  vector T(FCoupling_.T);
543  T += FContact_.T;
544 
545  // update body angular acceleration
546  alpha_ = inv(geomModel_->getI()) & T;
547  // update body angular velocity
548  vector Omega(Axis*omega + deltaT*alpha_);
549  // split Omega into Axis_ and omega_
550  omega_ = mag(Omega);
551 
552  if (omega_ < SMALL)
553  {
554  Axis_ = vector::one;
555  if (!case3D)
556  {
557  const vector validDirs = (geometricD + vector::one)/2;
558  Axis_ -= validDirs;
559  }
560  }
561  else
562  {
563  Axis_ = Omega/(omega_+SMALL);
564  if (!case3D)
565  {// in 2D, I need to keep only the part of the rotation axis
566  const vector validDirs = (geometricD + vector::one)/2;
567  Axis_ = cmptMultiply(vector::one-validDirs,Axis_);
568  }
569  }
570  Axis_ /= mag(Axis_);
571  }
572  };
573 
574  auto updateRotationFixedAxis = [&]()
575  {
576  vector T(FCoupling_.T);
577  T += FContact_.T;
578 
579  // update body angular velocity
580  vector Omega(Axis*omega + deltaT * (inv(geomModel_->getI()) & T));
581 
582  // split Omega into Axis_ and omega_
583  omega_ = mag(Omega);
584 
585  vector newAxis = Omega/(omega_+SMALL);
586  if ((newAxis & Axis_) < 0) Axis_ *= (-1.0);;
587  };
588 
589  if (bodyOperation_ == 0 or bodyOperation_ == 3)
590  {
591  return;
592  }
593  else if (bodyOperation_ == 1)
594  {
595  updateRotation();
596  return;
597  }
598  else if (bodyOperation_ == 2)
599  {
600  updateTranslation();
601  return;
602  }
603  else if (bodyOperation_ == 4)
604  {
605  updateRotationFixedAxis();
606  return;
607  }
608 
609  updateTranslation();
610  if (updateTorque_)
611  {
612  updateRotation();
613  }
614 
615  return;
616 }
617 //---------------------------------------------------------------------------//
618 // move immersed body according to body operation
619 void immersedBody::moveImmersedBody
620 (
621  scalar deltaT
622 )
623 {
624  if (bodyOperation_ == 0) return;
625 
626  if (geomModel_->getOwner() == Pstream::myProcNo())
627  {
628  if (mag(deltaT + 1.0) < SMALL) deltaT = mesh_.time().deltaT().value();
629 
630  // incremental rotation angle
631  scalar angle = omega_*deltaT;
632 
633  // translation increment
634  vector transIncr = Vel_*deltaT;
635 
636  // rotation matrix
637  tensor rotMatrix(Foam::cos(angle)*tensor::I);
638  rotMatrix += Foam::sin(angle)*tensor(
639  0.0, -Axis_.z(), Axis_.y(),
640  Axis_.z(), 0.0, -Axis_.x(),
641  -Axis_.y(), Axis_.x(), 0.0
642  );
643  rotMatrix += (1.0-Foam::cos(angle))*(Axis_ * Axis_);
644 
645  // update total rotation matrix
646  totRotMatrix_ = rotMatrix & totRotMatrix_;
647  vector eulerAngles;
648  scalar sy = Foam::sqrt(totRotMatrix_.xx()*totRotMatrix_.xx()
649  + totRotMatrix_.yy()*totRotMatrix_.yy());
650 
651  if (sy > SMALL)
652  {
653  eulerAngles.x() =
654  Foam::atan2(totRotMatrix_.zy(),totRotMatrix_.zz());
655  eulerAngles.y() = Foam::atan2(-totRotMatrix_.zx(),sy);
656  eulerAngles.z() =
657  Foam::atan2(totRotMatrix_.yx(),totRotMatrix_.xx());
658  }
659  else
660  {
661  eulerAngles.x() =
662  Foam::atan2(-totRotMatrix_.yz(),totRotMatrix_.yy());
663  eulerAngles.y() = Foam::atan2(-totRotMatrix_.zx(),sy);
664  eulerAngles.z() = 0.0;
665  }
666 
667  geomModel_->bodyRotatePoints(angle,Axis_);
668  geomModel_->bodyMovePoints(transIncr);
669  }
670 
671  geomModel_->synchronPos();
672 
673  InfoH << iB_Info;
674  InfoH << "-- body " << bodyId_ << " CoM : "
675  << geomModel_->getCoM() << endl;
676  InfoH << "-- body " << bodyId_ << " linear velocity : "
677  << Vel_ << endl;
678  InfoH << "-- body " << bodyId_ << " angluar velocity : "
679  << omega_ << endl;
680  InfoH << "-- body " << bodyId_ << " axis of rotation : "
681  << Axis_ << endl;
682  InfoH << "-- body " << bodyId_ << " total rotation matrix: "
683  << totRotMatrix_ << endl;
684 }
