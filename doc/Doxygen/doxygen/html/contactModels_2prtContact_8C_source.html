332  virtualMesh virtMesh(
333  *vmInfo,
334  cClass.getGeomModel(),
335  tClass.getGeomModel()
336  );
337 
338  scalar intersectedVolume(0);
339  vector contactCenter(vector::zero);
340  vector normalVector = vector::zero;
341  scalar contactArea(0);
342 
343  intersectedVolume = virtMesh.evaluateContact();
344 
345  if(virtMesh.getEdgeSVPoints().size() <= 4)
346  {
347  intersectedVolume = 0;
348  }
349 
350  if (intersectedVolume > 0)
351  {
352  Tuple2<scalar,vector> surfaceAndNormal =
353  virtMesh.get3DcontactNormalAndSurface
354  (

356  ||
357  tClass.getGeomModel().getcType() == nonConvex
358  );
359  contactArea = surfaceAndNormal.first();
360  normalVector = surfaceAndNormal.second();
361  contactCenter = virtMesh.getContactCenter();
362  }
363 
364  if (intersectedVolume > 0 && contactArea > 0)
365  {
366  subCInfo.getprtCntVars().contactCenter_ = contactCenter;
367  subCInfo.getprtCntVars().contactVolume_ = intersectedVolume;
368  subCInfo.getprtCntVars().contactNormal_ = normalVector;
369  subCInfo.getprtCntVars().contactArea_ = contactArea;
370  }
371  else
372  {
373  subCInfo.getprtCntVars().contactCenter_ = vector::zero;
374  subCInfo.getprtCntVars().contactVolume_ = 0;
375  subCInfo.getprtCntVars().contactNormal_ = vector::zero;
376  subCInfo.getprtCntVars().contactArea_ = 0;

379 //---------------------------------------------------------------------------//
380 void getPrtContactVars_Sphere
381 (
382  const fvMesh& mesh,
383  ibContactClass& cClass,
384  ibContactClass& tClass,
385  prtSubContactInfo& subCInfo
386 )
387 {
388  scalar cRadius = cClass.getGeomModel().getDC() / 2;
389  scalar tRadius = tClass.getGeomModel().getDC() / 2;
390  vector tCenter = tClass.getGeomModel().getCoM();
391 
392  vector centerDir = cClass.getGeomModel().getCoM()
393  - tCenter;
394 
395  scalar d = mag(centerDir);
396 
397  if(mag(centerDir) < SMALL || d > (cRadius + tRadius))
398  {
399  subCInfo.getprtCntVars().contactCenter_ = vector::zero;
400  subCInfo.getprtCntVars().contactVolume_ = 0;
401  subCInfo.getprtCntVars().contactNormal_ = vector::zero;
402  subCInfo.getprtCntVars().contactArea_ = 0;
403  return;
404  }
405 
406  scalar xLength = (sqr(d) - sqr(cRadius) + sqr(tRadius))/(2*d);
407 
408  if(sqr(xLength) > sqr(tRadius))
409  {
410  subCInfo.getprtCntVars().contactCenter_
411  = tCenter + (centerDir/d)*xLength;
412  subCInfo.getprtCntVars().contactVolume_
413  = (4/3)*Foam::constant::mathematical::pi*pow(tRadius,3);
414  subCInfo.getprtCntVars().contactNormal_ = centerDir/d;
415  subCInfo.getprtCntVars().contactArea_
416  = Foam::constant::mathematical::pi*sqr(tRadius);
417  return;
418  }
419 
420  if(case3D)
421  {
422 
423  scalar cSphCapV = (Foam::constant::mathematical::pi
424  *sqr(cRadius - d + xLength)
425  *(3*cRadius - (cRadius - d + xLength))) / 3;
426  scalar tSphCapV = (Foam::constant::mathematical::pi
427  *sqr(tRadius - xLength)
428  *(3*tRadius - (tRadius - xLength))) / 3;
429 
430  subCInfo.getprtCntVars().contactCenter_
431  = tCenter + (centerDir/d)*xLength;
432  subCInfo.getprtCntVars().contactVolume_
433  = cSphCapV + tSphCapV;
434  subCInfo.getprtCntVars().contactNormal_
435  = centerDir/d;
436  subCInfo.getprtCntVars().contactArea_
437  = Foam::constant::mathematical::pi
438  *sqr(sqrt(sqr(tRadius) - sqr(xLength)));
439  }
440  else
441  {
442  boundBox meshBounds = mesh.bounds();
443  scalar emptyLength = meshBounds.max()[emptyDim]
444  - meshBounds.min()[emptyDim];
445 
446  scalar cCirSeg = sqr(cRadius)*acos((d - xLength)/cRadius)
447  - (d - xLength)*sqrt(sqr(cRadius)
448  - sqr(d - xLength));
449  scalar tCirSeg = sqr(tRadius)*acos(xLength/tRadius)
450  - xLength*sqrt(sqr(tRadius) - sqr(xLength));
451 
452  subCInfo.getprtCntVars().contactCenter_
453  = tCenter + (centerDir/d)*xLength;
454  subCInfo.getprtCntVars().contactVolume_
455  = (cCirSeg + tCirSeg)*emptyLength;
456  subCInfo.getprtCntVars().contactNormal_
457  = centerDir/d;
458  subCInfo.getprtCntVars().contactArea_
459  = 2*sqrt(sqr(tRadius) - sqr(xLength))*emptyLength;
460  }
461 }

595 bool solvePrtContact(
596  const fvMesh& mesh,
597  prtContactInfo& cInfo,
598  prtSubContactInfo& subCInfo,
599  scalar deltaT
600 )
601 {
602  getPrtContactVars(
603  mesh,
604  cInfo.getcClass(),
605  cInfo.gettClass(),
606  subCInfo
607  );
608 
609  if (!(subCInfo.getprtCntVars().contactVolume_ > 0))
610  {
611  return false;
612  }
613 
614  InfoH << parallelDEM_Info << "-- Detected Particle-particle contact: -- body "
615  << subCInfo.getCPair().first() << " & -- body "
616  << subCInfo.getCPair().second() << endl;
617  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact cBody pos: "
618  << cInfo.getcClass().getGeomModel().getCoM() << " & tBody pos: "
619  << cInfo.gettClass().getGeomModel().getCoM() << endl;
620  InfoH << parallelDEM_Info << "-- body "<< subCInfo.getCPair().first() <<" linear velocity:"
621  << cInfo.getcVars().Vel_ << " magnitude: " << mag(cInfo.getcVars().Vel_) <<endl;
622  InfoH << parallelDEM_Info << "-- body "<< subCInfo.getCPair().first() <<" angular velocity:"
623  << cInfo.getcVars().omega_ << " magnitude: " << mag(cInfo.getcVars().omega_) <<endl;
624 
625  InfoH << parallelDEM_Info << "-- body "<< subCInfo.getCPair().second() <<" linear velocity:"
626  << cInfo.gettVars().Vel_ << " magnitude: " << mag(cInfo.gettVars().Vel_) <<endl;
627  InfoH << parallelDEM_Info << "-- body "<< subCInfo.getCPair().second() <<" angular velocity:"
628  << cInfo.gettVars().omega_ << " magnitude: " << mag(cInfo.gettVars().omega_) <<endl;
629  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact center "
630  << subCInfo.getprtCntVars().contactCenter_ << endl;
631  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact normal "
632  << subCInfo.getprtCntVars().contactNormal_ << endl;
633  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact volume "
634  << subCInfo.getprtCntVars().contactVolume_ << endl;
635  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact area "
636  << subCInfo.getprtCntVars().contactArea_ << endl;
637 
638  subCInfo.evalVariables(
639  cInfo.getcClass(),
640  cInfo.gettClass(),
641  cInfo.getcVars(),
642  cInfo.gettVars()
643  );
644 
645  // compute the normal force
646  vector F = subCInfo.getFNe();
647  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact FNe " << F << endl;
648 
649  vector FNd = subCInfo.getFNd();
650  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact FNd " << FNd << endl;
651 
652  // clamp FNd if opposite direction to FNe
653  if ((F & FNd) < 0 && mag(FNd) > mag(F))
654  {
655  FNd *= mag(F) / mag(FNd);
656  }
657 
658  F += FNd;
659  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact FN " << F << endl;
660 
661  vector Ft = subCInfo.getFt(deltaT);
662  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact Ft " << Ft << endl;
663 
664  if (mag(Ft) > cInfo.getMu() * mag(F))
665  {
666  Ft *= cInfo.getMu() * mag(F) / mag(Ft);
667  }
668  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact Ft clamped " << Ft << endl;
669  F += Ft;
670 
671  vector FA = subCInfo.getFA();
672  InfoH << parallelDEM_Info << "-- Particle-particle " <<subCInfo.getCPair().first() <<"-"<<subCInfo.getCPair().second() << " contact FA " << FA << endl;
673  F -= FA;
674 
675  // add the computed force to the affected bodies
676  subCInfo.getOutForce().first().F = F;
677  subCInfo.getOutForce().first().T = subCInfo.getcLVec() ^ F;
678  subCInfo.getOutForce().second().F = -F;
