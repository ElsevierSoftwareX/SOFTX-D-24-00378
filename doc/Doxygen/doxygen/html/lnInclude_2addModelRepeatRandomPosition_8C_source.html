267 bool addModelRepeatRandomPosition::shouldAddBody(const volScalarField& body)
268 {
269 
270  if (timeBased_)
271  {
272  scalar timeVal(mesh_.time().value());
273  scalar deltaTime(mesh_.time().deltaT().value());
274  scalar tmFrac(timeVal/timeBetweenUsage_);
275  tmFrac -= floor(tmFrac+deltaTime);
276 
277  InfoH << addModel_Info << "-- addModelMessage-- "
278  << "Time/(Time beween usage) - floor(Time/Time beween usage): "
279  << tmFrac << endl;
280 
281  InfoH << addModel_Info << "-- addModelMessage-- "
282  << "Number of bodies added on this time level: "
283  << addedOnTimeLevel_ << endl;
284 
285  bool tmLevelOk(tmFrac < deltaTime);
286 
287  if (not tmLevelOk)
288  {
289  addedOnTimeLevel_ = 0;
290  return false;
291  }
292 
293  if (partPerAdd_ <= addedOnTimeLevel_) {return false;}
294 
295  return (tmLevelOk and useNTimes_ > 0);
296  }
297 
298  if (fieldBased_)
299  {
300  scalar currentLambdaFrac(checkLambdaFraction(body));
301  if (currentLambdaFrac < fieldValue_ )
302  {
303  InfoH << addModel_Info << "-- addModelMessage-- "
304  << "Current lambda fraction = " << currentLambdaFrac
305  << " < then preset lambda fraction = " << fieldValue_ << endl;
306  return true;
307  }
308  }
309 
310  return false;
311

314 std::shared_ptr<geomModel> addModelRepeatRandomPosition::addBody
315 (
316  const volScalarField& body,
317  PtrList<immersedBody>& immersedBodies
318 )
319 {
320  geomModel_->resetBody();
321 
322  bodyAdditionAttemptCounter_++;
323 
324  // rotate
325  if (rotateParticles_)
326  {
327  scalar rotAngle = returnRandomAngle();
328  if (randomAxis_)
329  {
330  axisOfRot_ = returnRandomRotationAxis();
331  }
332  InfoH << addModel_Info << "-- addModelMessage-- "
333  << "Will rotate by " << rotAngle
334  << " PiRad around axis " << axisOfRot_ << endl;
335 
336  geomModel_->bodyRotatePoints(rotAngle,axisOfRot_);
337  //~ CoM = gSum(bodySurfMesh.coordinates())/bodySurfMesh.size();
338  }
339 
340  // scale
341  if (scaleApplication_ or scaleRandomApplication_)
342  {
343  if (scaleRandomApplication_){scaleStep_ = returnRandomScale();}
344  geomModel_->bodyScalePoints(scaleStep_);
345  }
346 
347  geomModel_->bodyScalePoints(1.02);
348 
349  vector CoM(geomModel_->getCoM());
350  point bBoxCenter = cellZoneBounds_.midpoint();
351  geomModel_->bodyMovePoints(bBoxCenter - CoM);
352 
353  // translate
354 
355  vector randomTrans = geomModel_->addModelReturnRandomPosition(allActiveCellsInMesh_,cellZoneBounds_,randGen_);
356  geomModel_->bodyMovePoints(randomTrans);
357 
358  // check if the body can be added
359  volScalarField helpBodyField_ = body;
360  geomModel_->createImmersedBody(
361  helpBodyField_,
362  octreeField_,
363  cellPoints_
364  );
365 
366  bool canAddBodyI = !isBodyInContact(immersedBodies);
367 
368  geomModel_->bodyScalePoints(1.0/1.02);
369 
370  reduce(canAddBodyI, andOp<bool>());
371  bodyAdded_ = (canAddBodyI);
372 
373  if(!bodyAdded_)
374  {
375  scaleCorrectionCounter_++;
376  }
377 
378  if(bodyAdded_)
379  {
380  if(timeBased_)
381  {
382  InfoH << addModel_Info << "-- addModelMessage-- "
383  << "addedOnTimeLevel: " << addedOnTimeLevel_<< endl;
384  addedOnTimeLevel_++;
385  InfoH << "-- addModelMessage-- " << "bodyAdded: "
386  << bodyAdded_ << " addedOnTimeLevel: " << addedOnTimeLevel_
387  << " useNTimes: " << useNTimes_<< endl;
388  if(addedOnTimeLevel_ == partPerAdd_)
389  {
390  useNTimes_--;
391  InfoH << "-- addModelMessage-- "
392  <<" useNTimes: " << useNTimes_<< endl;
393  reapeatedAddition_ = false;
394  }
395  }
396 
397  scaleCorrectionCounter_ = 0;
398 
399  }
400 
401  InfoH << addModel_Info << "-- addModelMessage-- "
402  << "bodyAdditionAttemptNr : " << bodyAdditionAttemptCounter_<< endl;
403  InfoH << "-- addModelMessage-- " << "sameScaleAttempts : "
404  << scaleCorrectionCounter_<< endl;
405 
406  if(scaleCorrectionCounter_ > nTriesBeforeScaling_ && scaleParticles_)
407  {
408  scaleApplication_ = true;
409  rescaleRequirement_ = true;
410  scalingFactor_++;
411  scaleStep_ = pow(scaleStep_,scalingFactor_);
412  if(scaleStep_<minScaleFit_)
413  {
414  scaleStep_=minScaleFit_;
415  }
416  scaleCorrectionCounter_ = 0;
417  }
418 
419  return geomModel_->getCopy();
420 }

521 //---------------------------------------------------------------------------//
522 labelList addModelRepeatRandomPosition::getBBoxCellsByOctTree
523 (
524  label cellToCheck,
525  bool& insideBB,
526  vector& bBoxMin,
527  vector& bBoxMax,
528  List<DynamicLabelList>& bBoxCells
529 )
530 {
531  labelList retList;
532 
533  if (octreeField_[cellToCheck] ==0)
534  {
535  octreeField_[cellToCheck] = 1;
536  vector cCenter = mesh_.C()[cellToCheck];
537  label partCheck(0);
538  forAll (bBoxMin,vecI)
539  {
540  if (cCenter[vecI] >= bBoxMin[vecI] and cCenter[vecI] <= bBoxMax[vecI])
541  {
542  partCheck++;
543  }
544  }
545  bool cellInside = (partCheck == 3) ? true : false;
546  if (cellInside)
547  {
548  bBoxCells[Pstream::myProcNo()].append(cellToCheck);
549  insideBB = true;
550  }
551  if (not insideBB or cellInside)
552  {
553  retList = mesh_.cellCells()[cellToCheck];
554  }
555  }
556  return retList;

559 scalar addModelRepeatRandomPosition::checkLambdaFraction(const volScalarField& body)
560 {
561  scalarList lambdaIntegrate(Pstream::nProcs());
562  scalarList volumeIntegrate(Pstream::nProcs());
563  scalar lambdaFraction(0);
564  forAll (lambdaIntegrate,k)
565  {
566  lambdaIntegrate[k] = 0;
567  volumeIntegrate[k] = 0;
568  }
569  forAll (cellsInBoundBox_[Pstream::myProcNo()],k)
570  {
571  label cell = cellsInBoundBox_[Pstream::myProcNo()][k];
572  lambdaIntegrate[Pstream::myProcNo()] += mesh_.V()[cell]*body[cell];
573  volumeIntegrate[Pstream::myProcNo()] += mesh_.V()[cell];
574  }
575  lambdaFraction = gSum(lambdaIntegrate)/gSum(volumeIntegrate);
576  InfoH << addModel_Info << "-- addModelMessage-- "
577  << "lambda fraction in controlled region: " << lambdaFraction<< endl;
578  return lambdaFraction;
579 }
