41 prtContactInfo::prtContactInfo
42 (
43  ibContactClass& cClass,
44  ibContactVars& cVars,
45  ibContactClass& tClass,
46  ibContactVars& tVars
47 )
48 :
49 cIbContactClass_(cClass),
50 cContactVars_(cVars),
51 tIbContactClass_(tClass),
52 tContactVars_(tVars)
53 {
54  contactPair_.first() = cVars.bodyId_;
55  contactPair_.second() = tVars.bodyId_;
56  const materialInfo& cMatInfo(cClass.getMatInfo());
57  const materialInfo& tMatInfo(tClass.getMatInfo());
58 
59  scalar adhPot = 0;
60  string adhPotKey;
61  if(cMatInfo.getMaterial() < tMatInfo.getMaterial())
62  {
63  adhPotKey = cMatInfo.getMaterial();
64  adhPotKey += "-";
65  adhPotKey += tMatInfo.getMaterial();
66  }
67  else
68  {
69  adhPotKey = tMatInfo.getMaterial();
70  adhPotKey += "-";
71  adhPotKey += cMatInfo.getMaterial();
72  }
73 
74  if(interAdhesion::getInterAdhesion().found(adhPotKey))
75  {
76  adhPot = interAdhesion::getInterAdhesion()[adhPotKey];
77  }
78 
79  // compute mean model parameters
80  physicalProperties_.aY_ = 1/((1 - sqr(cMatInfo.getNu()))/cMatInfo.getY()
81  + (1 - sqr(tMatInfo.getNu()))/tMatInfo.getY());
82  physicalProperties_.aG_ = 1/(2*(2 - cMatInfo.getNu())*(1 + cMatInfo.getNu())/cMatInfo.getY()
83  + 2*(2 - tMatInfo.getNu())*(1 + tMatInfo.getNu())/tMatInfo.getY());
84  physicalProperties_.aGammaN_ = (cMatInfo.getGamma()*tMatInfo.getGamma())
85  /(cMatInfo.getGamma()+tMatInfo.getGamma()+SMALL);
86  physicalProperties_.aGammat_ = (cMatInfo.getGamma()*tMatInfo.getGamma())
87  /(cMatInfo.getGamma()+tMatInfo.getGamma()+SMALL);
88  physicalProperties_.aMu_ = (cMatInfo.getMu()+tMatInfo.getMu())/2;
89  physicalProperties_.maxAdhN_ = cMatInfo.getAdhN() + tMatInfo.getAdhN() - 2*adhPot;
90  physicalProperties_.curAdhN_ = 0;
91  physicalProperties_.reduceM_ =
92  (
93  cIbContactClass_.getGeomModel().getM0()
94  *tIbContactClass_.getGeomModel().getM0()
95  /(cIbContactClass_.getGeomModel().getM0()
96  +tIbContactClass_.getGeomModel().getM0())
97  );
98  physicalProperties_.reduceBeta_ =
99  (
100  (-1)*sqrt(contactModelInfo::getBetaCoeff())*log((0.5*(cMatInfo.getEps()+tMatInfo.getEps())))/
101  (sqrt(sqr(log(cMatInfo.getEps()+tMatInfo.getEps()))+
102  sqr(Foam::constant::mathematical::pi)))
103  );
104 }

135 {
136  const boundBox cBBox(cIbContactClass_.getGeomModel().getBounds());
137  const boundBox tBBox(tIbContactClass_.getGeomModel().getBounds());
138  for (label coord = 0; coord < 3; coord++)
139  {
140  if (bbox.min()[coord] < cBBox.min()[coord])
141  {
142  bbox.min()[coord] = cBBox.min()[coord];
143  }
144 
145  if (bbox.max()[coord] > cBBox.max()[coord])
146  {
147  bbox.max()[coord] = cBBox.max()[coord];
148  }
149

151  {
152  bbox.min()[coord] = tBBox.min()[coord];
153  }
154 
155  if (bbox.max()[coord] > tBBox.max()[coord])
156  {
157  bbox.max()[coord] = tBBox.max()[coord];
158  }

179 void prtContactInfo::getContacts_ArbShape
180 (
181  scalar cellV
182 )
183 {
184  boundBox subCbBox(
185  cIbContactClass_.getGeomModel().getBounds().min(),
186  cIbContactClass_.getGeomModel().getBounds().max()
187  );
188 
189  if (!subCbBox.overlaps(tIbContactClass_.getGeomModel().getBounds()))
190  {
191  return;
192  }
193 
194  limitBBox(subCbBox);
195 
196  scalar charCellSize = pow(cellV,0.333333);
197  scalar subVolumeLength = charCellSize/virtualMeshLevel::getLevelOfDivision();
198  scalar subVolumeV = pow(subVolumeLength,3);
199 
200  newContactList_.emplace_back(matchSubContact(subCbBox, physicalProperties_, contactPair_));
201  newContactList_.back()->setVMInfo(subCbBox, subVolumeV);
202  return;
203 }

223 void prtContactInfo::syncContactList()
224 {
225  std::vector<std::shared_ptr<prtSubContactInfo>> syncedContactList;
226  // Sync the sub-contact list going through processors
227  for (label procI = 0; procI < Pstream::nProcs(); procI++)
228  {
229  label numOfCnts = 0;
230  if (procI == Pstream::myProcNo())
231  {
232  numOfCnts = contactList_.size();
233  }
234  reduce(numOfCnts, sumOp<label>());
235 
236  for (label i = 0; i < numOfCnts; i++)
237  {
238  bool vmInfoValid = false;
239  if (procI == Pstream::myProcNo())
240  {
241  vmInfoValid = contactList_[i]->getVMInfo() ? true : false;
242  }
243  reduce(vmInfoValid, orOp<bool>());
244 
245  if (vmInfoValid)
246  {
247  syncedContactList.emplace_back(std::make_shared<prtSubContactInfo>
248  (contactPair_, physicalProperties_)
249  );
250 
251  virtualMeshInfo vmInfoToSync;
252  if (procI == Pstream::myProcNo())
253  {
254  vmInfoToSync = virtualMeshInfo(*(contactList_[i]->getVMInfo()));
255  }
256  reduce(vmInfoToSync.sV.min(), sumOp<vector>());
257  reduce(vmInfoToSync.sV.max(), sumOp<vector>());
258  reduce(vmInfoToSync.subVolumeV, sumOp<scalar>());
259 
260  point startPointToReduce = vmInfoToSync.getStartingPoint();
261  if (procI != Pstream::myProcNo())
262  {
263  startPointToReduce = vector::zero;
264  }
265  reduce(startPointToReduce, sumOp<vector>());
266 
267  syncedContactList.back()->setVMInfo(vmInfoToSync);
268  }
269  else if (procI == 0)
270  {
271  syncedContactList.emplace_back(std::make_shared<prtSubContactInfo>
272  (contactPair_, physicalProperties_)
273  );
274  }
275  }
276  }
277 
278  contactList_.swap(syncedContactList);
279 }
