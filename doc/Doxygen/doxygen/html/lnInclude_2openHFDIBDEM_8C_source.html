617 //---------------------------------------------------------------------------//
618 void openHFDIBDEM::updateDEM(volScalarField& body,volScalarField& refineF)
619 {
620  if (cyclicPlaneInfo::getCyclicPlaneInfo().size() > 0)
621  {
622  forAll (immersedBodies_,bodyId)
623  {
624  if (!immersedBodies_[bodyId].getGeomModel().isCluster())
625  {
626  vector transVec = vector::zero;
627 
628  if (detectCyclicContact(
629  immersedBodies_[bodyId].getWallCntInfo(),
630  transVec
631  ))
632  {
633  verletList_.removeBodyFromVList(immersedBodies_[bodyId]);
634 
635  scalar thrSurf(readScalar(HFDIBDEMDict_.lookup("surfaceThreshold")));
636  std::shared_ptr<periodicBody> newPeriodicBody
637  = std::make_shared<periodicBody>(mesh_, thrSurf);
638 
639  newPeriodicBody->setRhoS(immersedBodies_[bodyId].getGeomModel().getRhoS());
640  std::shared_ptr<geomModel> iBcopy(immersedBodies_[bodyId].getGeomModel().getCopy());
641  iBcopy->bodyMovePoints(transVec);
642  newPeriodicBody->addBodyToCluster(immersedBodies_[bodyId].getGeomModelPtr());
643  newPeriodicBody->addBodyToCluster(iBcopy);
644  immersedBodies_[bodyId].getGeomModelPtr() = newPeriodicBody;
645 
646  verletList_.addBodyToVList(immersedBodies_[bodyId]);
647  Info << "Periodic body created for body " << bodyId << endl;
648  }
649  }
650  else
651  {
652  periodicBody& cBody = dynamic_cast<periodicBody&>(immersedBodies_[bodyId].getGeomModel());
653 
654  if(cBody.shouldBeUnclustered())
655  {
656  verletList_.removeBodyFromVList(immersedBodies_[bodyId]);
657 
658  immersedBodies_[bodyId].getGeomModelPtr() = cBody.getRemGeomModel();
659 
660  verletList_.addBodyToVList(immersedBodies_[bodyId]);
661  Info << "Periodic body unclustered for body " << bodyId << endl;
662  }
663  }
664  }
665  }
666 
667  scalar deltaTime(mesh_.time().deltaT().value());
668  scalar pos(0.0);
669  scalar step(stepDEM_);
670  // scalar timeStep(step*deltaTime);
671 
672  while( pos < 1)
673  {
674  InfoH << DEM_Info << " Start DEM pos: " << pos
675  << " DEM step: " << step << endl;
676 
677  InfoH << basic_Info << " DEM - CFD Time: "
678  << mesh_.time().value() + deltaTime*pos << endl;
679 
680  forAll (immersedBodies_,ib)
681  {
682  immersedBodies_[ib].updateMovement(deltaTime*step*0.5);
683  immersedBodies_[ib].moveImmersedBody(deltaTime*step);
684  }
685 
686  verletList_.update(immersedBodies_);
687 
688  DynamicLabelList wallContactIB;
689  DynamicList<wallSubContactInfo*> wallContactList;
690  forAll (immersedBodies_,bodyId)
691  {
692  immersedBody& cIb(immersedBodies_[bodyId]);
