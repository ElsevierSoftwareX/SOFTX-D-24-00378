76 void leastSquaresInt::imposeDirichletCondition
77 (
78  leastSquaresIntInfo& intpInfo,
79  volVectorField& Ui,
80  vectorField& ibPointsVal,
81  const Foam::fvMesh& mesh
82 )
83 {
84  const DynamicLabelList& ibCells = intpInfo.getSurfCells();
85  const List<point>& ibPoints = intpInfo.getIbPoints();
86  const List<vector>& ibNormals = intpInfo.getIbNormals();
87  const labelListList& cellCells = intpInfo.getCellCells();
88  const PtrList<scalarRectangularMatrix>& cInvDirMats =
89  intpInfo.getInvDirMats();
90 
91  autoPtr<vectorField> tpolyU(new vectorField(Ui.internalField(),ibCells));
92  vectorField& polyU = tpolyU();
93 
94  const vectorField& C = mesh_.cellCentres();
95 
96  label nCoeffs = 5;
97  if(case3D)
98  {
99  nCoeffs += 4;
100  }
101 
102  label counter = 0;
103  scalarField error(ibCells.size(), 0);
104  scalar maxError = 0;
105 
106  do
107  {
108  counter++;
109  scalar maxMagPolyU = 0;
110  forAll(ibCells, cellI)
111  {
112  label curCell = ibCells[cellI];
113  const labelList& curCells = cellCells[cellI];
114  const scalarRectangularMatrix& curInvMatrix = cInvDirMats[cellI];
115 
116  if(curCells.size() < nCoeffs)
117  {
118  Info << "Not eneough interp Points" << endl;
119  }
120  else
121  {
122  vectorField coeffs(nCoeffs, vector::zero);
123  vectorField source(curCells.size(), vector::zero);
124 
125  for(label i = 0; i < curCells.size(); i++)
126  {
127  source[i] = Ui[curCells[i]] - ibPointsVal[cellI];
128  }
129 
130  for(label i = 0; i < nCoeffs; i++)
131  {
132  for(label j = 0; j < source.size(); j++)
133  {
134  coeffs[i] += curInvMatrix[i][j]*source[j];
135  }
136  }
137 
138  vector oldPolyU = polyU[cellI];
139  vector R = C[curCell] - ibPoints[cellI];
140  if((R & ibNormals[cellI]) < 0)
141  R = vector::zero;
142 
143  if(case3D)
144  {
145  polyU[cellI] = ibPointsVal[cellI]
146  + coeffs[0]*R.x()
147  + coeffs[1]*R.y()
148  + coeffs[2]*R.x()*R.y()
149  + coeffs[3]*sqr(R.x())
150  + coeffs[4]*sqr(R.y())
151  + coeffs[5]*R.z()
152  + coeffs[6]*R.x()*R.z()
153  + coeffs[7]*R.y()*R.z()
154  + coeffs[8]*sqr(R.z());
155  }
156  else
157  {
158  labelList A (2,0.0);
159  label validDim = 0;
160  for(label dim = 0; dim < 3; dim++)
161  {
162  if(dim != emptyDim)
163  {
164  A[validDim++] = dim;
165  }
166  }
167  polyU[cellI] = ibPointsVal[cellI]
168  + coeffs[0]*R[A[0]]
169  + coeffs[1]*R[A[1]]
170  + coeffs[2]*R[A[0]]*R[A[1]]
171  + coeffs[3]*sqr(R[A[0]])
172  + coeffs[4]*sqr(R[A[1]]);
173  }
174 
175  error[cellI] = mag(polyU[cellI] - oldPolyU);
176  }
177  }
178 
179  forAll(ibCells, ci)
180  {
181  Ui[ibCells[ci]] = polyU[ci];
182  }
183 
184  if(!polyU.empty())
185  {
186  maxMagPolyU = max(mag(polyU));
187  }
188  else
189  {
190  maxMagPolyU = 0;
191  }
192 
193  error /= maxMagPolyU + SMALL;
194 
195  if(!polyU.empty())
196  {
197  maxError = max(error);
198  }
199  else
200  {
201  maxError = 0;
202  }
203 
204  } while(maxError > 0.01 && counter < 5);
205 }

208 (
209  leastSquaresIntInfo& intpInfo,

215  scalar fluxIn = 0;
216  scalar fluxOut = 0;

219  const labelList& ibFaces = intpInfo.getIbFaces();
220  const boolList& ibFaceFlips = intpInfo.getIbFacesFlips();

222  forAll (ibFaces, faceI)
223  {
224  const label curFace = ibFaces[faceI];
225  const bool curFlip = ibFaceFlips[faceI];
226 
227  const scalar curFlux = phi[curFace];
228 
229  if (!curFlip)
230  {
231  // Face points out of the live cell
232  if (curFlux >= 0)
233  {
234  // Flux out of the live cell
235  fluxOut += curFlux;
236  }
237  else
238  {
239  // Flux into the live cell
240  fluxIn -= curFlux;
241  }
242  }
243  else
244  {
245  // Face points into the live cell: flip it
246  if (curFlux >= 0)
247  {
248  // Flux into of the live cell
249  fluxIn += curFlux;
250  }
251  else
252  {
253  // Flux out the live cell
254  fluxOut -= curFlux;
255  }
256  }
257  }
258 
259  scalar imbalance = fluxIn - fluxOut;
260 
261  Info<< " fluxIn = " << fluxIn << " fluxOut = " << fluxOut
262  << " imbalance = " << imbalance
263  << endl;
264 
265  scalar massCorr = 1.0;
266 
267  if (mag(imbalance) > SMALL)
268  {
269  // Scaling required: scale to match the smaller of two
270  // fluxes
271  if (fluxIn > fluxOut)
272  {
273  // Scale down incoming flux
274  // Note change of sign: imbalance is negative
275  massCorr = 1 - imbalance/(fluxIn + SMALL);
276 
277  Info<< "Scaling down incoming flux with factor = "
278  << massCorr << endl;
279 
280  scalar newFluxIn = 0;
281 
282  // Visit all incoming flux faces and re-scale the flux
283  forAll (ibFaces, faceI)
284  {
285  const label curFace = ibFaces[faceI];
286  const bool curFlip = ibFaceFlips[faceI];
287 
288  if (mesh_.isInternalFace(curFace))
289  {
290  // Take reference to current flux
291  scalar& curFlux = phi[curFace];
292 
293  if (!curFlip)
294  {
295  // Face points out of the live cell
296  if (curFlux < 0)
297  {
298  // Flux out of the live cell
299  curFlux *= massCorr;
300  newFluxIn -= curFlux;
301  }
302  }
303  else
304  {
305  // Face points into the live cell: flip it
306  if (curFlux >= 0)
307  {
308  // Flux out the live cell
309  curFlux *= massCorr;
310  newFluxIn += curFlux;
311  }
312  }
313  }
314  }
315  }
316  else
317  {
318  // Scale down outgoing flux
319  massCorr = 1 + imbalance/(fluxOut + SMALL);
320 
321  Info<< "Scaling down outgoing flux with factor = "
322  << massCorr << endl;
323 
324  scalar newFluxOut = 0;
325 
326  // Visit all outgoing flux faces and re-scale the flux
327  forAll (ibFaces, faceI)
328  {
329  const label curFace = ibFaces[faceI];
330  const bool curFlip = ibFaceFlips[faceI];
331 
332  if (mesh_.isInternalFace(curFace))
333  {
334  // Take reference to current flux
