208 vector wallSubContactInfo::getFt(wallContactVars& wallCntvar, scalar deltaT)
209 {
210  physicalProperties& meanCntPar(wallCntvar.getMeanCntPar());
211  // project last Ft into a new direction
212  vector FtLastP(wallCntvar.FtPrev_
213  - (wallCntvar.FtPrev_ & wallCntvar.contactNormal_)
214  *wallCntvar.contactNormal_);
215  // scale projected Ft to have same magnitude as FtLast
216  // vector FtLastS(mag(wallCntvar.FtPrev_) * (FtLastP/(mag(FtLastP)+SMALL)));
217  // compute relative tangential velocity
218  vector cVeliNorm = wallCntvar.Veli_
219  - ((wallCntvar.Veli_ & wallCntvar.contactNormal_)
220  *wallCntvar.contactNormal_);
221 
222  vector Vt(cVeliNorm - vector::zero);
223  // compute tangential force

226  wallCntvar.FtPrev_ = - meanCntPar.aG_*wallCntvar.Lc_*Vt*deltaT + Ftdi;
