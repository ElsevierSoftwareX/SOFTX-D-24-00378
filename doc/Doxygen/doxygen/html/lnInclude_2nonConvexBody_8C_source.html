38 labelList nonConvexBody::getBBoxCellsByOctTree
39 (

41  bool& insideBB,
42  vector& bBoxMin,
43  vector& bBoxMax,
44  List<DynamicLabelList>& bBoxCells,
45  Field<label>& octreeField
46 )
47 {
48  labelList retList;
49 
50  if (octreeField[cellToCheck] ==0)

54  label partCheck(0);
55  forAll (bBoxMin,vecI)

61  }

63  if (cellInside)

66  insideBB = true;
67  }
68  if (not insideBB or cellInside)

70  retList = mesh_.cellCells()[cellToCheck];
71  }
72  }
73  return retList;

76 // create immersed body for convex body
77 void nonConvexBody::createImmersedBody
78 (
79  volScalarField& body,
80  Field<label>& octreeField,
81  List<labelList>& cellPoints
82 )
83 {
84  // clear old list contents
85  intCells_[Pstream::myProcNo()].clear();
86  surfCells_[Pstream::myProcNo()].clear();
87 
88  // reduce computational domain to the body bounding box
89  scalar inflFact(2*sqrt(mesh_.magSf()[0]));
90 
91  boundBox ibBound(getBounds());
92 
93  vector expMinBBox = ibBound.min() - vector::one*inflFact;
94  vector expMaxBBox = ibBound.max() + vector::one*inflFact;
95 
96  octreeField *= 0;
97  List<DynamicLabelList> bBoxCells(Pstream::nProcs());
98 
99  bool isInsideBB(false);
100  labelList nextToCheck(1,0);
101  label iterCount(0);label iterMax(mesh_.nCells());
102  while ((nextToCheck.size() > 0 or not isInsideBB) && iterCount < iterMax)
103  {
104  iterCount++;
105  DynamicLabelList auxToCheck;
106 
107  forAll (nextToCheck,cellToCheck)
108  {
109  auxToCheck.append(
110  getBBoxCellsByOctTree(
111  nextToCheck[cellToCheck],
112  isInsideBB,
113  expMinBBox,
114  expMaxBBox,
115  bBoxCells,
116  octreeField
117  )
118  );
119  }
120  nextToCheck = auxToCheck;
121  }
122 
123  // get cell centers inside the body bounding box
124  const pointField& cp = mesh_.C();
125  const pointField fCp = filterField(cp,bBoxCells[Pstream::myProcNo()]);
126  boolList fCentersInside = triSurfSearch_().calcInside(fCp);
127 
128  // find the processor with most of this IB inside
129  ibPartialVolume_[Pstream::myProcNo()] = 0;
130 
131  vector sDSpan(4.0*(mesh_.bounds().max()-mesh_.bounds().min()));
132 
133  // first loop, construction of body field and identification of
134  // the number of inside and surface cells
135  const pointField& pp = mesh_.points();
136  forAll (bBoxCells[Pstream::myProcNo()],bCellI) //go only through bBox
137  {
138  label cellI(bBoxCells[Pstream::myProcNo()][bCellI]);
139 
140  // check if partially or completely inside
141  const pointField vertexPoints(pp,cellPoints[cellI]);
142  boolList vertexesInside = triSurfSearch_().calcInside( vertexPoints );
143  bool centerInside(fCentersInside[bCellI]);
144  scalar rVInSize(0.5/vertexesInside.size());
145  // Note: weight of a single vertex in the cell
146 
147  scalar cBody(0);
148  forAll (vertexesInside, verIn)
149  {
150  if (vertexesInside[verIn]==true)
151  {
152  cBody += rVInSize; //fraction of cell covered
153  }
154  }
155 
156  if (centerInside)
157  {
158  cBody+=0.5;
159  }
160  if (cBody > thrSurf_)
161  {
162  if (cBody > (1.0-thrSurf_))
163  {
164  intCells_[Pstream::myProcNo()].append(cellI);
165  cellToStartInCreateIB_ = cellI;
166  }
167  else if (cBody <= (1.0-thrSurf_))
168  {
169  surfCells_[Pstream::myProcNo()].append(cellI);
170  if (sdBasedLambda_)
171  {
172  pointIndexHit pointHit(
173  triSurfSearch_().nearest(
174  mesh_.C()[cellI],
175  sDSpan
176  )
177  );
178  scalar signedDist(0.0);
179  if (pointHit.hit())
180  {
181  signedDist = mag(pointHit.hitPoint()-cp[cellI]);
182  }
183  else
184  {
185  InfoH << iB_Info
186  << "Missed point in signedDist computation !!"
187  << endl;
188  }
189  if (centerInside)
190  {
191  cBody = 0.5*(Foam::tanh(intSpan_*signedDist/Foam::pow(mesh_.V()[cellI],0.333))+1.0);
192  }
193  else
194  {
195  cBody = 0.5*(-1.0*Foam::tanh(intSpan_*signedDist/Foam::pow(mesh_.V()[cellI],0.333))+1.0);
196  }
197  }
198  }
199  ibPartialVolume_[Pstream::myProcNo()] += 1;
200  }
201  body[cellI]+= cBody;
202  // clip the body field values
203  body[cellI] = min(max(0.0,body[cellI]),1.0);
204  }
205 
206  // gather partial volume from other processors
207  Pstream::gatherList(ibPartialVolume_, 0);
208  Pstream::scatter(ibPartialVolume_, 0);
209 
210  for (label i = 0; i < ibPartialVolume_.size(); i++)
211  {
212  if (ibPartialVolume_[i] == max(ibPartialVolume_))
213  {
214  //Set owner of the IB which will move this IB
215  owner_ = i;
216  break;
217  }
218  }
219 }
