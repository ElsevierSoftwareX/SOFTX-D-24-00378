129 void leastSquaresIntInfo::getInvDirichletMatrix
130 (
131  const DynamicLabelList& ibCells
132 )
133 {
134  const List<point>& ibPoints = getIbPoints();
135  //const List<vector>& ibNormals = getIbNormals();
136  const labelListList& cellCells = getCellCells();
137  resetInvDirMat(ibCells.size());
138  PtrList<scalarRectangularMatrix>& cInvDirMats = getInvDirMats();
139  const vectorField& C = mesh_.cellCentres();
140 
141  label nCoeffs = 5;
142  if(case3D)
143  {
144  nCoeffs += 4;
145  }
146 
147  forAll(ibCells, cellI)
148  {
149  const labelList& curCells = cellCells[cellI];
150 
151  vectorField allPoints(curCells.size(), vector::zero);
152 
153  if(allPoints.size() < nCoeffs)
154  {
155  Info << "Not eneough interp Points" << endl;
156  }
157  else
158  {
159  forAll(curCells, cI)
160  {
161  allPoints[cI] = C[curCells[cI]];
162  }
163 
164  //Weights
165  point origin = C[ibCells[cellI]];
166  scalarField curDist = mag(allPoints - origin);
167  scalarField W = 0.5*
168  (
169  1+cos(Foam::constant::mathematical::pi*curDist/
170  (1.1*max(curDist)))
171  );
172 
173  cInvDirMats.set
174  (
175  cellI,
176  new scalarRectangularMatrix
177  (
178  nCoeffs,
179  allPoints.size(),
180  0.0
181  )
182  );
183  scalarRectangularMatrix& curMatrix = cInvDirMats[cellI];
184  scalarRectangularMatrix Mi
185  (
186  allPoints.size(),
187  nCoeffs,
188  0.0
189  );
190 
191  origin = ibPoints[cellI];
192 
193  for(label i = 0; i < allPoints.size(); i++)
194  {
195  scalar X = allPoints[i].x() - origin.x();
196  scalar Y = allPoints[i].y() - origin.y();
197  scalar Z = allPoints[i].z() - origin.z();
198  if(case3D)
199  {
200  label coeff = 0;
201  Mi[i][coeff++] = X;
202  Mi[i][coeff++] = Y;
203  Mi[i][coeff++] = X*Y;
204  Mi[i][coeff++] = sqr(X);
205  Mi[i][coeff++] = sqr(Y);
206  Mi[i][coeff++] = Z;
207  Mi[i][coeff++] = X*Z;
208  Mi[i][coeff++] = Y*Z;
209  Mi[i][coeff++] = sqr(Z);
210  }
211  else
212  {
213  labelList A (2,0.0);
214  scalarList dists(3,0.0);
215  dists[0] = X;
216  dists[1] = Y;
217  dists[2] = Z;
218  label validDim = 0;
219  for(label dim = 0; dim < 3; dim++)
220  {
221  if(dim != emptyDim)
222  {
223  A[validDim++] = dim;
224  }
225  }
226  label coeff = 0;
227  Mi[i][coeff++] = dists[A[0]];
228  Mi[i][coeff++] = dists[A[1]];
229  Mi[i][coeff++] = dists[A[0]]*dists[A[1]];
230  Mi[i][coeff++] = sqr(dists[A[0]]);
231  Mi[i][coeff++] = sqr(dists[A[1]]);
232  }
233  }
234 
235  for(label i = 0; i < Mi.m(); i++)
236  {
237  for(label j = 0; j < Mi.n(); j++)
238  {
239  Mi[i][j] *= W[i];
240  }
241  }
242 
243  scalarSquareMatrix lsM(nCoeffs, 0.0);
244  for(label i = 0; i < lsM.m(); i++)
245  {
246  for(label j = 0; j < lsM.n(); j++)
247  {
248  for(label k = 0; k < Mi.m(); k++)
249  {
250  lsM[i][j] += Mi[k][i]*Mi[k][j];
251  }
252  }
253  }
254 
255  scalarSquareMatrix invLsM = LUinvert(lsM);
256 
257  for(label i = 0; i < lsM.m(); i++)
258  {
259  for(label j = 0; j < Mi.m(); j++)
260  {
261  for(label k = 0; k < lsM.m(); k++)
262  {
263  curMatrix[i][j] += invLsM[i][k]*Mi[j][k]*W[j];
264  }
265  }
266  }
267  }
268  }

313 void leastSquaresIntInfo::findCellCells
314 (
315  const label cellId,
316  labelList& cellCells,
317  scalar& centerMeanDist
318 )
319 {
320  labelHashSet cellSet;
321  cellSet.insert(cellId);
322  labelList currCells;
323  labelList auxCells = cellSet.toc();
324 
325  for(label nRow = 0; nRow < maxCCRows_; nRow++)
326  {
327  currCells = auxCells;
328  auxCells.clear();
329  forAll(currCells, cellI)
330  {
331  label curCell = currCells[cellI];
332  const labelList& curCellPoints = mesh_.cellPoints()[curCell];
333  forAll(curCellPoints, pointI)
334  {
335  label curPoint = curCellPoints[pointI];
336  const labelList& curPointCells = mesh_.pointCells()[curPoint];
337  forAll(curPointCells, nCellI)
338  {
339  label nCellId = curPointCells[nCellI];
340  if(!cellSet.found(nCellId))
341  {
342  cellSet.insert(nCellId);
343  auxCells.append(nCellId);
344  }
345  }
346  }
347  }
348 
349  if (nRow == 0 && auxCells.size())
350  {
351  const vectorField& C = mesh_.cellCentres();
352  forAll(auxCells, cellI)
353  {
354  centerMeanDist += mag(C[cellId] - C[auxCells[cellI]]);
355  }
356  centerMeanDist /= auxCells.size();
357  }
358  }
359 
360  cellSet.erase(cellId);
361  cellCells = cellSet.toc();
362 }
