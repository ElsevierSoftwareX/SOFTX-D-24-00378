42 wallContactInfo::wallContactInfo
43 (
44  ibContactClass& cClass,
45  ibContactVars& cVars
46 )
47 :
48 ibContactClass_(cClass),
49 ibContactVars_(cVars)
50 {

52  const materialInfo& cMatInfo(cClass.getMatInfo());
53 
54  const List<string> cntPatches = wallMatInfo::getWallPatches();
55  forAll(cntPatches, patchI)
56  {
57  const materialInfo& wInfo = wallMatInfo::getWallMatInfo()[cntPatches[patchI]];
58 
59  scalar adhPot = 0;
60  string adhPotKey;
61 
62  if(cMatInfo.getMaterial() < wInfo.getMaterial())
63  {
64  adhPotKey = cMatInfo.getMaterial();
65  adhPotKey += "-";
66  adhPotKey += wInfo.getMaterial();
67  }
68  else
69  {
70  adhPotKey = wInfo.getMaterial();
71  adhPotKey += "-";
72  adhPotKey += cMatInfo.getMaterial();
73  }
74  if(interAdhesion::getInterAdhesion().found(adhPotKey))
75  {
76  adhPot = interAdhesion::getInterAdhesion()[adhPotKey];
77  }
78 
79  // compute mean model parameters
80  scalar aY = 1/((1 - sqr(cMatInfo.getNu()))/cMatInfo.getY()
81  + (1 - sqr(wInfo.getNu()))/wInfo.getY());
82  scalar aG = 1/(2*(2 - cMatInfo.getNu())*(1 + cMatInfo.getNu())
83  /cMatInfo.getY() + 2*(2 - wInfo.getNu())
84  *(1 + wInfo.getNu())/wInfo.getY());
85  scalar aGammaN = (cMatInfo.getGamma()*wInfo.getGamma())
86  /(cMatInfo.getGamma()+wInfo.getGamma()+SMALL);
87  scalar aGammat = (cMatInfo.getGamma()*wInfo.getGamma())
88  /(cMatInfo.getGamma()+wInfo.getGamma()+SMALL);
89  scalar aMu = (cMatInfo.getMu()+wInfo.getMu())/2;
90  scalar maxAdhN = cMatInfo.getAdhN() + wInfo.getAdhN() - 2*adhPot;
91  scalar reduceBeta =
92  (
93  (-1*sqrt(contactModelInfo::getBetaCoeff()))*log((0.5*(cMatInfo.getEps()+wInfo.getEps())))/
94  (sqrt(sqr(log(cMatInfo.getEps()+wInfo.getEps()))+
95  sqr(Foam::constant::mathematical::pi)))
96  );
97  wallMeanPars_.insert(
98  cntPatches[patchI],
99  physicalProperties(aY, aG, aGammaN, aGammat, aMu, maxAdhN, 0, 0, reduceBeta)
100  );
101  // Info << "reduceBeta " <<reduceBeta <<endl;
102  // Info << "sqrt(5) "<< sqrt(5) <<endl;
103  }

339 List<DynamicList<vector>> wallContactInfo::detectPossibleSMContact
340 (
341  DynamicVectorList& contactPoints,
342  List<string>& contactPatches
343 )
344 {
345  vectorHashSet possibleContactElements;
346  vectorHashSet checkedOctreeFaces;
347  List<DynamicVectorList> baseSubContactList;
348  label iterMax(SM_().matrixSize_[0]*SM_().matrixSize_[1]*SM_().matrixSize_[2]);
349 
350  autoPtr<DynamicVectorList> contactElements(
351  new DynamicVectorList);
352 
353  // InfoH << DEM_Info <<" -- ConstructSM - Is Allive : #1.1 " << endl;
354 
355  forAll(contactPoints,cP)
356  {
357  vector elementIndex(SM_().getSMCentroidIndex(contactPoints[cP]));
358  if(!possibleContactElements.found(elementIndex))
359  {
360  possibleContactElements.insert(elementIndex);
361  SM_()[elementIndex].stlFound = true;
362  SM_()[elementIndex].initPointSet = true;
363  SM_()[elementIndex].initPoint = contactPoints[cP];
364  contactElements().append(elementIndex);
365  }
366  }
367  // InfoH << DEM_Info <<" -- ConstructSM - Is Allive : #1.2 " << endl;
368  autoPtr<DynamicVectorList> nextToCheck(
369  new DynamicVectorList);
370 
371  autoPtr<DynamicVectorList> auxToCheck(
372  new DynamicVectorList);
373 
374  autoPtr<DynamicVectorList> subContactElements(
375  new DynamicVectorList);
376 
377  label iterCount2(0);
378  // InfoH << DEM_Info <<" -- ConstructSM - Is Allive : #1.3 " << endl;
379  while((contactElements().size() > SMALL) and iterCount2++ < iterMax)
380  {
381  label iterCount(0);
382  nextToCheck().clear();
383  subContactElements().clear();
384 
385  nextToCheck().append(contactElements()[0]);
386  // nextToCheck().append(edgeFaceNeighbours(mesh,commonFaces[0]));
387  while ((nextToCheck().size() > 0) and iterCount++ < iterMax)
388  {
389  auxToCheck().clear();
390  forAll(nextToCheck(),cE)
391  {
392  if (SM_()[nextToCheck()[cE]].toCheck)
393  {
394  checkSMElement(nextToCheck()[cE],contactPatches);
395  bool isInMesh(false);
396  bool isInBody(false);
397  bool allInMesh(false);
398  bool allInBody(false);
399  checkElement(nextToCheck()[cE],isInMesh,isInBody,allInMesh,allInBody);
400  if(isInBody && isInMesh)
401  {
402  if(!allInMesh)
403  {
404  subContactElements().append(nextToCheck()[cE]);
405  auxToCheck().append(SM_().faceNeighbourElements(nextToCheck()[cE]));
406  checkedOctreeFaces.insert(nextToCheck()[cE]);
407  if(allInBody)
408  {
409  SM_()[nextToCheck()[cE]].isInternal = allInBody;
410  }
411  }
412  }
413  if((isInBody && !isInMesh)|| SM_()[nextToCheck()[cE]].stlFound)
414  {
415  subContactElements().append(nextToCheck()[cE]);
416  checkedOctreeFaces.insert(nextToCheck()[cE]);
417  auxToCheck().append(SM_().faceNeighbourElements(nextToCheck()[cE]));
418  if(allInBody)
419  {
420  SM_()[nextToCheck()[cE]].isInternal = allInBody;
421  }
422  }
423  }
424  }
425  const autoPtr<DynamicVectorList> helpPtr(nextToCheck.ptr());
426  nextToCheck.set(auxToCheck.ptr());
427  auxToCheck = helpPtr;
428  }
429  if(subContactElements().size() > 0)
430  {
431  baseSubContactList.append(subContactElements());
432  }
433  for (auto it = contactElements().begin(); it != contactElements().end();)
434  {
435  if (checkedOctreeFaces.found(*it))
436  {
437  it = contactElements().erase(it);
438  }
439  else
440  {
441  ++it;
442  }
443  }
444  }
445  return baseSubContactList;
446 }

525

530  SM_()[index].isMesh = isMeshLocC;
531 
532  const List<vector> verticesLabels = SM_().elementVertexIndexies(index);
533  forAll(verticesLabels, vL)

539  SM_()(verticesLabels[vL]).toCheck = false;
540  SM_()(verticesLabels[vL]).isCBody = ibContactClass_.getGeomModel().pointInside(vertexPoint);
541  bool isMeshLocV(true);

544  isMeshLocV *= isInsidePlane(vertexPoint,contactPatches[iC]);

553 void wallContactInfo::checkElement
554 (
555  vector& index,
556  bool& inMesh,
557  bool& inBody,
558  bool& allInMesh,
559  bool& allInBody
560 )
561 {
562  bool inMeshLocal(false);
563  bool inBodyLocal(false);
564  bool allInMeshLocal(true);
565  bool allInBodyLocal(true);
566 
567  inMeshLocal = inMeshLocal || SM_()[index].isMesh;
568  allInMeshLocal *= SM_()[index].isMesh;
569  inBodyLocal = inBodyLocal || SM_()[index].isCBody;
570  allInBodyLocal *= SM_()[index].isCBody;
571  if(SM_()[index].isCBody && !SM_()[index].initPointSet)
572  {
573  SM_()[index].initPoint = SM_()[index].center;
574  SM_()[index].initPointSet = true;
575  }
576  const List<vector> verticesLabels = SM_().elementVertexIndexies(index);
577 
578  forAll(verticesLabels, vL)
579  {
580  inMeshLocal = inMeshLocal || SM_()(verticesLabels[vL]).isMesh;
581  allInMeshLocal *= SM_()(verticesLabels[vL]).isMesh;
582  inBodyLocal = inBodyLocal || SM_()(verticesLabels[vL]).isCBody;
583  allInBodyLocal *= SM_()(verticesLabels[vL]).isCBody;
584  if(SM_()(verticesLabels[vL]).isCBody && !SM_()[index].isCBody && !SM_()[index].initPointSet)
585  {
586  SM_()[index].initPoint = SM_()(verticesLabels[vL]).center;
587  SM_()[index].initPointSet = true;
588  }
589  }
590 
591  inMesh = inMeshLocal;
592  inBody = inBodyLocal;
593  allInMesh = allInMeshLocal;
594  allInBody = allInBodyLocal;
595 }
