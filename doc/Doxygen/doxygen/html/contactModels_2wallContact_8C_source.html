236 )

240  scalar contactArea(0);
241  vector contactNormal(vector::zero);
242 
243  autoPtr<DynamicVectorList> contactCenters(
244  new DynamicVectorList);
245 
246  autoPtr<DynamicVectorList> contactPlaneCenters(
247  new DynamicVectorList);
248 
249  autoPtr<DynamicScalarList> contactAreas(
250  new DynamicScalarList);
251 
252  label vMContactInfoSize = sCW.getVMContactSize();
253  label vMPlaneInfoSize = sCW.getVMPlaneSize();
254  const List<Tuple2<point,boundBox>>& sCInternalInfo = sCW.getInternalElements();
255  const List<string>& contactPatches = sCW.getContactPatches();
256 
257  for(label i = 0; i< vMContactInfoSize; i++)
258  {
259  autoPtr<virtualMeshWallInfo> vmWInfo = sCW.getVMContactInfo(i);
260  if (!vmWInfo.valid())
261  {
262  continue;
263  }
264 
265  virtualMeshWall virtMeshWall(
266  vmWInfo(),
267  wallCntInfo.getcClass().getGeomModel()
268  );
269 
270  if(virtMeshWall.detectFirstContactPoint())
271  {
272  intersectVolume += virtMeshWall.evaluateContact();
273  contactCenters().append(virtMeshWall.getContactCenter());
274  }
275  }
276 
277  forAll(sCInternalInfo,sCII)
278  {
279  intersectVolume += sCInternalInfo[sCII].second().volume();
280  contactCenters().append(sCInternalInfo[sCII].first());
281  }
282 
283  if(intersectVolume>0)
284  {
285  for(label i = 0; i< vMPlaneInfoSize; i++)
286  {
287  autoPtr<virtualMeshWallInfo> vmWInfo = sCW.getVMPlaneInfo(i);
288  if (!vmWInfo.valid())
289  {
290  continue;
291  }
292 
293  autoPtr<virtualMeshWall> virtMeshPlane(new virtualMeshWall(
294  vmWInfo(),
295  wallCntInfo.getcClass().getGeomModel()
296  ));
297 
298  if(virtMeshPlane->detectFirstFaceContactPoint())
299  {
300  scalar contactAreaLoc = (virtMeshPlane->evaluateContact()/vmWInfo->getSVVolume())*(pow(vmWInfo->getSVVolume(),2.0/3));
301  contactAreas().append(contactAreaLoc);
302  contactPlaneCenters().append(virtMeshPlane->getContactCenter());
303  }
304  else
305  {
306  contactAreas().append((intersectVolume/vmWInfo->getSVVolume())*(pow(vmWInfo->getSVVolume(),2.0/3)));
307  contactPlaneCenters().append(virtMeshPlane->getContactCenter());
308  }
309  }
310 
311  forAll(contactCenters(),cC)
312  {
313  contactCenter += contactCenters()[cC];
314  }
315  // Pout << "contactCenter " << contactCenter.size() << endl;
316  contactCenter /= contactCenters().size();
317  // Pout << "Survived #0 " << endl;

319  forAll(contactAreas(),cA)
320  {
321  contactArea += contactAreas()[cA];
322  }
323 
324  if(contactArea == 0)

326  return;
327  }
328  // Pout << "contactArea " << contactArea << endl;
329  // Pout << "Survived #0.5 " << endl;
330  forAll(contactPatches,cP)
331  {
332  contactNormal -= wallPlaneInfo::getWallPlaneInfo()[contactPatches[cP]][0]*contactAreas()[cP];
333  }
334  // Pout << "contactNormal " << mag(contactNormal) << endl;
335  contactNormal /=mag(contactNormal);
336  // Pout << "contactNormal " << contactNormal << endl;
337  wallCntInfo.getcClass().setWallContact(true);
338  wallCntInfo.getcClass().inContactWithStatic(true);
339  // Pout << "Survived #1 " << endl;
340  wallContactVars& wallCntVars = sCW.getWallCntVars();
341  // Pout << "Survived #2 " << endl;
342  wallCntVars.contactCenter_ = contactCenter;
343  wallCntVars.contactArea_ = contactArea;
344  wallCntVars.contactVolume_ = intersectVolume;
345  wallCntVars.contactNormal_ = contactNormal;
346  // Pout << "Survived #3 " << endl;
347  wallCntVars.setMeanCntPars_Plane
348  (
349  contactAreas(),
350  contactPatches,
351  wallCntInfo.getWallMeanPars()
352  );
353  // Pout << "Survived #4 " << endl;
354  }

357 void getWallContactVars_Sphere(
358  const fvMesh& mesh,
359  wallContactInfo& wallCntInfo,
360  const scalar deltaT,
361  wallSubContactInfo& sCW
362 )
363 {
364  vector cCenter(wallCntInfo.getcClass().getGeomModel().getCoM());
365 
366  List<string>& contactPatches = wallCntInfo.getContactPatches();
367 
368  List<wallContactVars> wallCntVarsList;
369  forAll(contactPatches, patchI)
370  {
371  List<vector> planeInfo = wallPlaneInfo::getWallPlaneInfo()[contactPatches[patchI]];
372  plane p(planeInfo[1], planeInfo[0]);
373  point nearestPoint = p.nearestPoint(cCenter);
374 
375  vector nVec = (cCenter - nearestPoint)/mag(cCenter - nearestPoint);
376 
377  wallCntVarsList.append(wallContactVars());
378 
379  wallCntVarsList.last().contactCenter_ = nearestPoint;
380  wallCntVarsList.last().contactNormal_ = nVec;
381  wallCntVarsList.last().contactArea_ = sphereContactArea
382  (
383  mesh,
384  wallCntInfo.getcClass(),
385  wallCntInfo.getcVars(),
386  p
387  );
388 
389  wallCntVarsList.last().contactVolume_ = getInterVolume_Sphere
390  (
391  mesh,
392  wallCntInfo.getcClass(),
393  wallCntInfo.getcVars(),
394  p
395  );
396  }
397 
398  wallContactVars& wallCntVars = sCW.getWallCntVars();
399 
400  List<scalar> contactAreas;
401 
402  for (int i = 0; i < wallCntVarsList.size(); ++i)
403  {
404  wallCntVars.contactCenter_ += wallCntVarsList[i].contactCenter_ * wallCntVarsList[i].contactVolume_;
405  wallCntVars.contactNormal_ += wallCntVarsList[i].contactNormal_ * wallCntVarsList[i].contactVolume_;
406  wallCntVars.contactArea_ += wallCntVarsList[i].contactArea_;
407  contactAreas.append(wallCntVarsList[i].contactArea_);
408  wallCntVars.contactVolume_ += wallCntVarsList[i].contactVolume_;
409  }
410 
411  wallCntVars.contactCenter_ /= wallCntVars.contactVolume_;
412  wallCntVars.contactNormal_ /= wallCntVars.contactVolume_;
413 
414  wallCntVars.setMeanCntPars_Plane
415  (
416  contactAreas,
417  contactPatches,
418  wallCntInfo.getWallMeanPars()
419  );
420 
421  if(wallCntVars.contactVolume_ > 0)
422  {
423  wallCntInfo.getcClass().setWallContact(true);
424  wallCntInfo.getcClass().inContactWithStatic(true);
425  }
426 }

541  cLVecOut += wallCntVar.lVec_;

548 scalar getInterVolume_Sphere(
549  const fvMesh& mesh,
550  ibContactClass& cClass,
551  ibContactVars& cVars,
552  plane& cPlane
553 )
554 {
555  scalar cRadius(cClass.getGeomModel().getDC() / 2);
556  vector cCenter(cClass.getGeomModel().getCoM());
557 
558  scalar dist = cPlane.distance(cCenter);
559  scalar xH = cRadius - dist;
560 
561  if(case3D)
562  {
563  return (Foam::constant::mathematical::pi*sqr(xH)*(3*cRadius - xH)) / 3;
564  }
565  else
566  {
567  boundBox meshBounds = mesh.bounds();
568  scalar emptyLength = meshBounds.max()[emptyDim]
569  - meshBounds.min()[emptyDim];
570 
571  scalar cCirSeg = sqr(cRadius)*acos((dist)/cRadius)
572  - (dist)*sqrt(sqr(cRadius) - sqr(dist));
573  return cCirSeg*emptyLength;
574  }
575 }
