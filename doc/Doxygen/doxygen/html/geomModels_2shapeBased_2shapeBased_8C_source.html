47 vector shapeBased::addModelReturnRandomPosition
48 (
49  const bool allActiveCellsInMesh,
50  const boundBox cellZoneBounds,
51  Random& randGen
52 )
53 {
54  vector ranVec(vector::zero);
55 
56  meshSearch searchEng(mesh_);
57 
58  // get its center of mass
59  vector CoM(getCoM());
60 
61  const vector validDirs = (geometricD + vector::one)/2;
62  vector dirCorr(cmptMultiply((vector::one - validDirs),CoM));
63  dirCorr += cmptMultiply((vector::one - validDirs),0.5*(mesh_.bounds().max() + mesh_.bounds().min()));
64 
65  // get the body boundBox
66  boundBox bodyBounds(getBounds());
67  // compute the max scales to stay in active bounding box
68  vector maxScales(cellZoneBounds.max() - bodyBounds.max());
69  maxScales -= cellZoneBounds.min() - bodyBounds.min();
70  maxScales *= 0.5*0.9;//0.Y is there just to be sure
71 
72  InfoH << addModel_Info << "-- addModelMessage-- "
73  << "acceptable movements: " << maxScales << endl;
74 
75  scalar ranNum = 0;
76  for (int i=0;i<3;i++)
77  {
78  ranNum = 2.0*maxScales[i]*randGen.scalar01() - 1.0*maxScales[i];
79  ranVec[i] = ranNum;
80  }
81 
82  ranVec = cmptMultiply(validDirs,ranVec); //translate only with respect to valid directions
83  ranVec += dirCorr;
84 
85  return ranVec;
86 }
