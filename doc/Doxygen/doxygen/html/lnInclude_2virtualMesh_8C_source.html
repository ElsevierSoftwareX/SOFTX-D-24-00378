205  {
206  cVolumeTypeMixed = true;
207

245  )
246  );
247

250  contactVolume += crossBBox.volume();

253  }
254  return;
255  }

261

337 Tuple2<scalar,vector> virtualMesh::get3DcontactNormalAndSurface(DynamicPointList edgeSubVolumesPoints)
338 {
339 // This function is taken from prtContact and just adjustated for higher accuracy
340  scalar area(0.0);
341  vector normalVec(vector::zero);
342  scalar tDC(tGeomModel_.getDC());
343  vector normalVector = vector::zero;
344  point closestPoint = vector::zero;
345 
346  point subContactCenter = vector::zero;
347  forAll (edgeSubVolumesPoints,cell)
348  {
349  subContactCenter += edgeSubVolumesPoints[cell];
350  }
351  subContactCenter /= edgeSubVolumesPoints.size();
352 
353  tGeomModel_.getClosestPointAndNormal
354  (
355  subContactCenter,
356  vector::one * tDC,
357  closestPoint,
358  normalVector
359  );
360 
361  if (edgeSubVolumesPoints.size() >= 3)
362  {
363  bool normOk(false);
364  vector center(vector::zero);
365 
366  forAll (edgeSubVolumesPoints,cell)
367  {
368  center += edgeSubVolumesPoints[cell];
369  }
370  center /= edgeSubVolumesPoints.size();
371 
372  scalar xx(0);
373  scalar xy(0);
374  scalar xz(0);
375  scalar yy(0);
376  scalar yz(0);
377  scalar zz(0);
378 
379  forAll (edgeSubVolumesPoints,cell)
380  {
381  vector subPoint(edgeSubVolumesPoints[cell] - center);
382  if(subPoint != vector::zero)
383  subPoint = subPoint/mag(subPoint);
384  xx += subPoint[0] * subPoint[0];
385  xy += subPoint[0] * subPoint[1];
386  xz += subPoint[0] * subPoint[2];
387  yy += subPoint[1] * subPoint[1];
388  yz += subPoint[1] * subPoint[2];
389  zz += subPoint[2] * subPoint[2];
390  }
391 
392  xx /= edgeSubVolumesPoints.size();
393  xy /= edgeSubVolumesPoints.size();
394  xz /= edgeSubVolumesPoints.size();
395  yy /= edgeSubVolumesPoints.size();
396  yz /= edgeSubVolumesPoints.size();
397  zz /= edgeSubVolumesPoints.size();
398 
399  vector weightedDir(vector::zero);
400 
401  scalar detX(yy*zz-yz*yz);
402  vector axisDirX(detX,xz*yz-xy*zz,xy*yz-xz*yy);
403  scalar weightX(detX*detX);
404  if((weightedDir & axisDirX) < 0.0)
405  weightX = -weightX;
406  weightedDir += axisDirX * weightX;
407 
408  scalar detY(xx*zz-xz*xz);
409  vector axisDirY(xz*yz-xy*zz,detY,xy*xz-yz*xx);
410  scalar weightY(detY*detY);
411  if((weightedDir & axisDirY) < 0.0)
412  weightY = -weightY;
413  weightedDir += axisDirY * weightY;
414 
415  scalar detZ(xx*yy-xy*xy);
416  vector axisDirZ(xy*yz-xz*yy,xy*xz-yz*xx,detZ);
417  scalar weightZ(detZ*detZ);
418  if((weightedDir & axisDirZ) < 0.0)
419  weightZ = -weightZ;
420  weightedDir += axisDirZ * weightZ;
421 
422  if(mag(weightedDir) > SMALL)
423  {
424  normOk = true;
425  normalVec = weightedDir/mag(weightedDir);
426  }
427  if (!normOk || mag(normalVec) < 1)
428  normalVec = normalVector;
429 
430  // create best fitting plane
431  plane bestFitPlane(contactCenter_, normalVec);
432  normalVec = bestFitPlane.normal();
433  DynamicPointList commCellsPosInPlane;
434  forAll (edgeSubVolumesPoints,cell)
435  {
436  commCellsPosInPlane.append(bestFitPlane.nearestPoint(edgeSubVolumesPoints[cell]));
437  }
438 
439  vector q1(1.0, 0.0, 0.0);
440  vector q2(0.0, 1.0, 0.0);
441  if (abs(q1 & bestFitPlane.normal()) > abs(q2 & bestFitPlane.normal()))
442  q1 = q2;
443 
444  vector u(bestFitPlane.normal() ^ q1);
445  vector v(bestFitPlane.normal() ^ u);
446 
447  DynamicList<plane> clockwisePlanes;
448  List<scalar> helpList(6);
449  helpList[0] = 0.0;
450  helpList[1] = 1.0;
451  helpList[2] = 0.0;
452  helpList[3] = -1.0;
453  helpList[4] = 0.0;
454  helpList[5] = 1.0;
455 
456  // loop over parts of plane to find and sort points
457  DynamicVectorList commCellsInSections;
458  for (label i = 0; i < 4; i++)
459  //~ for (label i = 0; i < 8; i++)
460  {
461  scalar uStep(helpList[i + 1] - helpList[i]);
462  scalar vStep(helpList[i + 2] - helpList[i + 1]);
463  DynamicPointList pointsInSection;
464  //~ for (scalar j = 0.0; j < 5.0; j += 1.0)
465  for (scalar j = 0.0; j < 15.0; j += 1.0)
466  {
467  plane uPlane(contactCenter_, u*(helpList[i] + uStep*j/15.0) + v*(helpList[i + 1] + vStep*j/15.0));
468  plane vPlane(contactCenter_, u*(helpList[i] + uStep*(j+1)/15.0) + v*(helpList[i + 1] + vStep*(j+1)/15.0));
469 
470  forAll (commCellsPosInPlane, celli)
471  {
472  if (uPlane.sideOfPlane(commCellsPosInPlane[celli]) == 0 && vPlane.sideOfPlane(commCellsPosInPlane[celli]) == 1)
473  {
474  pointsInSection.append(commCellsPosInPlane[celli]);
475  }
476  }
477 
478  if (pointsInSection.size() > SMALL)
479  {
480  vector max(vector::zero);
481  scalar distance(0);
482  forAll (pointsInSection, pointI)
483  {
484  scalar magnitude = mag(pointsInSection[pointI]-contactCenter_);
485  if (magnitude > distance)
486  {
487  magnitude = distance;
488  max = pointsInSection[pointI];
489  }
490  }
491  commCellsInSections.append(max);
492  }
493  }
494  }
495  // calculate contact area
496  for (label i = 0; i + 1 < commCellsInSections.size(); i++)
497  {
498  //~ Info << commCellsInSections[i] << endl;
499  vector AC(commCellsInSections[i] - contactCenter_);
500  vector BC(commCellsInSections[i + 1] - contactCenter_);
501  vector crossPr(AC ^ BC);
502  area += mag(crossPr)/2;
503  }
504 
505  if (commCellsInSections.size() > 2)
506  {
507  vector AC(commCellsInSections[commCellsInSections.size() - 1] - contactCenter_);
508  vector BC(commCellsInSections[0] - contactCenter_);
509  vector crossPr(AC ^ BC);
510  area += mag(crossPr)/2;
511  }
512  }
513 
514  if (normalVec == vector::zero)
515  {
516  normalVec = normalVector;
517  }
518 
519  if ((normalVector & normalVec) < 0)
520  {
521  normalVec = normalVec * (-1);
522  }
523 
524  Tuple2<scalar,vector> returnValue(area,normalVec);
525 
526  return returnValue;
527 }
