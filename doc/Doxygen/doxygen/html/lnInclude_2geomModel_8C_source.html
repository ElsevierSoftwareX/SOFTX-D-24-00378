100 void geomModel::addToMAndI
101 (
102  volScalarField& body,
103  DynamicLabelList& labelCellLst
104 )
105 {
106  forAll (labelCellLst,cell)
107  {
108  label cellI = labelCellLst[cell];
109 
110  scalar Mi = body[cellI]*rhoS_.value()*mesh_.V()[cellI];
111  // add to M_
112 
113  M_ += Mi;
114  scalar xLoc = mesh_.C()[cellI].x() - CoM_.x();
115  scalar yLoc = mesh_.C()[cellI].y() - CoM_.y();
116  scalar zLoc = mesh_.C()[cellI].z() - CoM_.z();
117 
118  // add to I_
119  I_.xx() += Mi*(yLoc*yLoc + zLoc*zLoc);
120  I_.yy() += Mi*(xLoc*xLoc + zLoc*zLoc);
121  I_.zz() += Mi*(xLoc*xLoc + yLoc*yLoc);
122 
123  I_.xy() -= Mi*(xLoc*yLoc);
124  I_.xz() -= Mi*(xLoc*zLoc);
125  I_.yz() -= Mi*(yLoc*zLoc);
126  }
127 }

219 void geomModel::correctSurfCells
220 (
221  volScalarField& body,
222  DynamicLabelList& potentSurfCells,
223  HashTable<bool, label, Hash<label>>& cellInside,
224  List<labelList>& cellPoints
225 )
226 {
227  HashTable<bool, label, Hash<label>> verticesStatus(potentSurfCells.size()*6);
228  forAll(potentSurfCells, cellLabel)
229  {
230  label cCell = potentSurfCells[cellLabel];
231  scalar cBody(0);
232 
233  if(cellInside[cCell])
234  {
235  cBody += 0.5;
236  }
237 
238  const labelList& cVerts = cellPoints[cCell];
239  scalar rVInSize(0.5/cVerts.size());
240  forAll(cVerts, vertI)
241  {
242  if(!verticesStatus.found(cVerts[vertI]))
243  {
244  bool vertexInside = pointInside(mesh_.points()[cVerts[vertI]]);
245  verticesStatus.insert
246  (
247  cVerts[vertI],
248  vertexInside
249  );
250 
251  if(vertexInside)
252  {
253  cBody += rVInSize;
254  }
255  }
256  else if(verticesStatus[cVerts[vertI]])
257  {
258  cBody += rVInSize;
259  }
260  }
261 
262  if (cBody > thrSurf_)
263  {
264  if (cBody > (1.0-thrSurf_))
265  {
266  intCells_[Pstream::myProcNo()].append(cCell);
267  }
268  else
269  {
270  surfCells_[Pstream::myProcNo()].append(cCell);
271  }
272  ibPartialVolume_[Pstream::myProcNo()] += 1;
273 
274  body[cCell] += cBody;
275 
276  // clip the body field values
277  body[cCell] = min(max(0.0,body[cCell]),1.0);
278  }
279  }
280 }
