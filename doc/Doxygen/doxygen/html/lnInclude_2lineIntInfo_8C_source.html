103 void lineIntInfo::correctIntPoint
104 (
105  point ibPoint,
106  intPoint& cPoint
107 )
108 {
109  if(cPoint.iProc_ != Pstream::myProcNo())
110  {
111  return;
112  }
113 
114  vector closestPoint = getClosestPoint(ibPoint, cPoint);
115 
116  if(pointInCell(closestPoint, cPoint.iCell_))
117  {
118  cPoint.iPoint_ = closestPoint;
119  }
120  else
121  {
122  const labelList& cellFaces(mesh_.cells()[cPoint.iCell_]);
123 
124  forAll (cellFaces, fi)
125  {
126  const face faceI = mesh_.faces()[cellFaces[fi]];
127  vector dir = closestPoint - cPoint.iPoint_;
128 
129  pointHit pHit = faceI.ray(
130  cPoint.iPoint_,
131  dir,
132  mesh_.points()
133  );
134 
135  if(pHit.hit())
136  {
137  vector newP = 0.95*(pHit.hitPoint() - cPoint.iPoint_);
138  newP += cPoint.iPoint_;
139 
140  if(pointInCell(newP, cPoint.iCell_))
141  {
142  cPoint.iPoint_ = newP;
143  break;
144  }
145  }
146  }

164 intPoint lineIntInfo::findIntPoint
165 (
166  intPoint& fromP,
167  point& endP
168 )
169 {
170  if(fromP.iPoint_ == endP)
171  {
172  return intPoint();
173  }
174 
175  intPoint retP
176  (
177  endP,
178  fromP.iCell_,
179  fromP.iProc_
180  );
181 
182  if(fromP.iProc_ == Pstream::myProcNo())
183  {
184  label faceInDir = -1;
185  while(!pointInCell(retP.iPoint_, retP.iCell_))
186  {
187  faceInDir = getFaceInDir(retP, faceInDir);
188  if (!mesh_.isInternalFace(faceInDir))
189  {
190  label facePatchId(mesh_.boundaryMesh().whichPatch(faceInDir));
191  const polyPatch& cPatch = mesh_.boundaryMesh()[facePatchId];
192 
193  if (cPatch.type() == "processor")
194  {
195  const processorPolyPatch& procPatch
196  = refCast<const processorPolyPatch>(cPatch);
197 
198  label sProc = (Pstream::myProcNo() == procPatch.myProcNo())
199  ? procPatch.neighbProcNo() : procPatch.myProcNo();
200 
201  retP.iCell_ = cPatch.whichFace(faceInDir);
202  retP.iProc_ = sProc;
203 
204  return retP;
205  }
206  else
207  {
208  retP.iProc_ = -1;
209  return retP;
210  }
211  }
212 
213  label owner(mesh_.owner()[faceInDir]);
214  label neighbour(mesh_.neighbour()[faceInDir]);
215  retP.iCell_ = (retP.iCell_ == neighbour) ? owner : neighbour;
216  }
217 
218  return retP;
219  }
220 
221  return retP;
222 }

224 label lineIntInfo::getFaceInDir
225 (
226  const intPoint& retPoint,
227  const label prevFace
228 )
229 {
230  label faceToReturn = -1;
231  vector dir = retPoint.iPoint_ - mesh_.C()[retPoint.iCell_];
232 
233  const labelList& cellFaces(mesh_.cells()[retPoint.iCell_]);
234  scalar dotProd(-GREAT);
235 
236  forAll (cellFaces, faceI)
237  {
238  label fI = cellFaces[faceI];
239  if(fI != prevFace)
240  {
241  vector outNorm = (mesh_.faceOwner()[fI] == retPoint.iCell_)
242  ? mesh_.Sf()[fI] : (-1*mesh_.Sf()[fI]);
243 
244  scalar auxDotProd(outNorm & dir);
245  if (auxDotProd > dotProd)
246  {
247  dotProd = auxDotProd;
248  faceToReturn = fI;
249  }
250  }
251  }
252 
253  return faceToReturn;
254 }

256 bool lineIntInfo::pointInCell
257 (
258  point pToCheck,
259  label cToCheck
260 )
261 {
262  const labelList& cellFaces(mesh_.cells()[cToCheck]);
263  forAll(cellFaces, faceI)
264  {
265  label fI = cellFaces[faceI];
266  vector outNorm = mesh_.Sf()[fI];
267  outNorm = (mesh_.faceOwner()[fI] == cToCheck) ? outNorm : (-1*outNorm);
268 
269  if (((pToCheck - mesh_.Cf()[fI]) & outNorm) > 0)
270  {
271  return false;
272  }
273  }
274  return true;
