197 label virtualMeshWall::getInternalSV()
198 {
199  autoPtr<DynamicVectorList> nextToCheck(
200  new DynamicVectorList);
201  autoPtr<DynamicVectorList> auxToCheck(
202  new DynamicVectorList);
203 
204  nextToCheck->append(bbMatrix_.getSVIndexForPoint(vMeshWallInfo_.getStartingPoint()));
205  label innerSVCount(0);
206  vectorHashSet octreeSvSet;
207 
208  while (nextToCheck->size() > 0)
209  {
210  auxToCheck().clear();
211  forAll (nextToCheck(),sV)
212  {
213  if (!octreeSvSet.found(nextToCheck()[sV]))
214  {
215  octreeSvSet.insert(nextToCheck()[sV]);
216  subVolumeProperties& cSubVolume = bbMatrix_[nextToCheck()[sV]];
217  if (cSubVolume.isCBody)
218  {
219  bool isNotOnEdge(true);
220  List<vector> neighbourSubVolumes = bbMatrix_.faceNeighbourSubVolumes(nextToCheck()[sV]);
221  neighbourSubVolumes.append(bbMatrix_.edgeNeighbourSubVolumes(nextToCheck()[sV]));
222  neighbourSubVolumes.append(bbMatrix_.cornerNeighbourSubVolumes(nextToCheck()[sV]));
223  forAll(neighbourSubVolumes,nSV)
224  {
225  isNotOnEdge *= bbMatrix_[neighbourSubVolumes[nSV]].isCBody;
226  }
227  if(!isNotOnEdge)
228  {
229  cSubVolume.isOnEdge = true;
230  }
231  auxToCheck().append(bbMatrix_.faceNeighbourSubVolumes(nextToCheck()[sV]));
232  }
233  if(!cSubVolume.isOnEdge)
234  {
235  innerSVCount++;
236  }
237  }
238  }
239 
240  const autoPtr<DynamicVectorList> helpPtr(nextToCheck.ptr());
241  nextToCheck.set(auxToCheck.ptr());
242  auxToCheck = helpPtr;
243  }
244  return innerSVCount;
245 }
