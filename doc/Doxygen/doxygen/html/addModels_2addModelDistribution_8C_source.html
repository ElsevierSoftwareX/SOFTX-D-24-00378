113  cellZonePoints_.setSize(Pstream::nProcs());

116

123  {
124  fieldValue_ = (readScalar(addModeICoeffs_.lookup("fieldValue")));
125  fieldBased_ = true;

127  << "addModel will control particles volume fraction" << endl;
128  InfoH << "-- addModelMessage-- " << "preset volume fraction: "
129  << fieldValue_ << endl;
130  }
131  else
132  {
133  InfoH << addModel_Info << "-- addModelMessage-- "
134  << "notImplemented, will crash" << endl;
135  }
136 
137  if (addDomain_ == "cellZone")
138  {
139  zoneName_ = (word(addDomainCoeffs_.lookup("zoneName")));
140  cellZoneActive_ = true;
141  initializeCellZone();
142  InfoH << addModel_Info << "-- addModelMessage-- "
143  << "cellZone based addition zone" << endl;
144  }
145  else if (addDomain_ == "boundBox")
146  {
147  minBound_ = (addDomainCoeffs_.lookup("minBound"));
148  maxBound_ = (addDomainCoeffs_.lookup("maxBound"));
149  boundBoxActive_ = true;
150 
151  initializeBoundBox();
152  InfoH << addModel_Info << "-- addModelMessage-- "
153  << "boundBox based addition zone" << endl;
154  }
155  else if (addDomain_ == "domain")
156  {
157  InfoH << addModel_Info << "-- addModelMessage-- "
158  << "notImplemented, will crash" << endl;
159  }
160  else
161  {
162  InfoH << addModel_Info << "-- addModelMessage-- "
163  << "notImplemented, will crash" << endl;
164  }
165 
166  // check, if the whole zone is in the mesh
167  scalarList procZoneVols(Pstream::nProcs());
168  procZoneVols[Pstream::myProcNo()] = 0;
169  forAll (cellsInBoundBox_[Pstream::myProcNo()],cellI)
170  {
171  procZoneVols[Pstream::myProcNo()]+=mesh_.V()[cellsInBoundBox_[Pstream::myProcNo()][cellI]];
172  }
173 
174  Pstream::gatherList(procZoneVols, 0);
175  Pstream::scatter(procZoneVols, 0);
176 
177  scalar zoneVol(0);
178  forAll (procZoneVols, procI)
179  {
180  zoneVol += procZoneVols[procI];

188  << "addition zone NOT completely immersed in mesh "

190  InfoH << zoneVol << " " << zoneBBoxVol << endl;
191  }
192  else
193  {
194  InfoH << addModel_Info << "-- addModelMessage-- "
195  << "addition zone completely immersed in mesh -> OK" << endl;
196  }
197 
198  partPerAddTemp_ = partPerAdd_;

203 {

212  InfoH << addModel_Info << "-- addModelMessage-- "

215 
216  InfoH << "-- addModelMessage-- "
217  << "Number of bodies added on this time level: "
218  << addedOnTimeLevel_ << endl;
219 
220  bool tmLevelOk(tmFrac < deltaTime);
221 
222  if (not tmLevelOk)
223  {
224  addedOnTimeLevel_ = 0;
225  return false;
226  }
227 
228  if (partPerAdd_ <= addedOnTimeLevel_) {return false;}

452 scalar addModelDistribution::checkLambdaFraction(const volScalarField& body)
453 {
454  scalarList lambdaIntegrate(Pstream::nProcs());
455  scalarList volumeIntegrate(Pstream::nProcs());
456  scalar lambdaFraction(0);
457  forAll (lambdaIntegrate,k)
458  {
459  lambdaIntegrate[k] = 0;
460  volumeIntegrate[k] = 0;
461  }
462  forAll (cellsInBoundBox_[Pstream::myProcNo()],k)
463  {
464  label cell = cellsInBoundBox_[Pstream::myProcNo()][k];
465  lambdaIntegrate[Pstream::myProcNo()] += mesh_.V()[cell]*body[cell];
466  volumeIntegrate[Pstream::myProcNo()] += mesh_.V()[cell];
467  }
468  lambdaFraction = gSum(lambdaIntegrate)/gSum(volumeIntegrate);
469  InfoH << addModel_Info << "-- addModelMessage-- "
470  << "lambda fraction in controlled region: " << lambdaFraction<< endl;
471  return lambdaFraction;

496 Tuple2<label, scalar> addModelDistribution::returnScaleFactor()
497 {
498  DynamicLabelList missingParticles;
499  scalar maxSizeVolume = 1.0/6.0*3.14*pow(stlBaseSize_
500  *particleSize_[particleSize_.size()-1]
501  *convertToMeters_/stlBaseSize_,3);
502  forAll (addedParticlesSize_,size)
503  {
504  scalar distribDiff = distribution_[size] - 100*addedParticlesSize_[size]/(volumeOfAddedBodies_+SMALL);
505  if(distribDiff > 0)
506  {
507  scalar meanFactor = particleSize_[size]*convertToMeters_/stlBaseSize_;
508  scalar meanVolume = 1.0/6.0*3.14*pow(stlBaseSize_ * meanFactor,3);
509  missingParticles.append(floor((distribDiff*maxSizeVolume)/meanVolume));
510  }
511  else
512  {
513  missingParticles.append(0);
514  }
515  }
516 
517  label totalMissParts(0);
518  forAll (missingParticles,size)
519  {
520  totalMissParts += missingParticles[size];
521  }
522 
523  label randomMissPart = floor(randGen_.scalar01()*totalMissParts);
524  label missingPart(0);
525  forAll (missingParticles,size)
526  {
527  missingPart = size;
528  randomMissPart -= missingParticles[size];
529  if(randomMissPart < 0)
530  {
531  break;
532  }
533  }
534 
535  scalar factor(particleSize_[missingPart - 1] + (particleSize_[missingPart] - particleSize_[missingPart - 1]) * randGen_.scalar01());
536  factor *= convertToMeters_/stlBaseSize_;
537 
538  Tuple2<label, scalar> returnValue(missingPart, factor);
539 
540  return returnValue;
