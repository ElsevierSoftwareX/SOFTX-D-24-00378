65 void lineInt::correctVelocity
66 (
67  lineIntInfo& intpInfo,
68  volVectorField& Ui,
69  vectorField& ibPointsVal,
70  const Foam::fvMesh& mesh
71 )
72 {
73  const DynamicLabelList& cSurfCells = intpInfo.getSurfCells();
74 
75  List<point>& ibPoints = intpInfo.getIbPoints();
76  List<List<intPoint>>& intPoints = intpInfo.getIntPoints();
77 
78  getCurVelocity(intPoints);
79  List<label> intOrder = getIntOrder(intPoints);
80 
81  forAll(intPoints, ibp)
82  {
83  label cellI = cSurfCells[ibp];
84 
85  switch(intOrder[ibp])
86  {
87  case 0:
88  {
89  Ui[cellI] = ibPointsVal[ibp];
90  break;
91  }
92 
93  case 1:
94  {
95  vector VP1 = intPoints[ibp][0].iVel_ - ibPointsVal[ibp];
96 
97  // distance between interpolation points
98  scalar deltaR = mag(intPoints[ibp][0].iPoint_ - ibPoints[ibp]);
99 
100  // cell center to surface distance
101  scalar ds = mag(mesh.C()[cellI] - ibPoints[ibp]);
102 
103  vector linCoeff = VP1/(deltaR+SMALL);
104 
105  Ui[cellI] = linCoeff*ds + ibPointsVal[ibp];
106  break;
107  }
108 
109  case 2:
110  {
111  vector VP1 = intPoints[ibp][0].iVel_ - ibPointsVal[ibp];
112  vector VP2 = intPoints[ibp][1].iVel_ - ibPointsVal[ibp];
113 
114  // distance between interpolation points
115  scalar deltaR1 = mag(intPoints[ibp][0].iPoint_ - ibPoints[ibp]);
116  scalar deltaR2 = mag(intPoints[ibp][1].iPoint_
117  - intPoints[ibp][0].iPoint_);
118 
119  // cell center to surface distance
120  scalar ds = mag(mesh.C()[cellI] - ibPoints[ibp]);
121 
122  vector quadCoeff = (VP2 - VP1)*deltaR1 - VP1*deltaR2;
123  quadCoeff /= (deltaR1*deltaR2*(deltaR1 + deltaR2)+SMALL);
124 
125  vector linCoeff = (VP1-VP2)*Foam::pow(deltaR1,2.0);
126  linCoeff += 2.0*VP1*deltaR1*deltaR2;
127  linCoeff += VP1*Foam::pow(deltaR2,2.0);
128  linCoeff /= (deltaR1*deltaR2*(deltaR1 + deltaR2)+SMALL);
129 
130  Ui[cellI] = quadCoeff*ds*ds + linCoeff*ds + ibPointsVal[ibp];
131  break;
132  }
133  }
134  }
135 }
