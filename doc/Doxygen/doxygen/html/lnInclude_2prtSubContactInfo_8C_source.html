110 vector prtSubContactInfo::getFt(scalar deltaT)
111 {
112  // compute relative tangential velocity
113  vector cVeliNorm = cVeli_ - ((cVeli_ & prtCntVars_.contactNormal_)
114  *prtCntVars_.contactNormal_);
115 
116  vector tVeliNorm = tVeli_ - ((tVeli_ & prtCntVars_.contactNormal_)
117  *prtCntVars_.contactNormal_);
118 
119  vector Vt(cVeliNorm - tVeliNorm);
120  // compute tangential force
121  vector Ftdi(- physicalProperties_.aGammat_
122  *sqrt(physicalProperties_.aG_*physicalProperties_.reduceM_*Lc_)*Vt);
123  Ftdi -= physicalProperties_.aG_*Lc_*Vt*deltaT;
124  return Ftdi;
125 }
